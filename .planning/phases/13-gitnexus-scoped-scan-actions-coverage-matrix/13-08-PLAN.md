```ts
// Repo-segment regex: must start with [a-z0-9], then allow [a-z0-9\-_.] (D-13-EXT-09a).
// Followed by .refine() that catches `.` and `..` segments (which the regex alone allows
// as a single-char `.` or `[a-z0-9]` followed by `.` etc). Refine is the defence-in-depth
// layer — the regex is the first wall, refine is the second.
const RepoTargetRe = /^[a-z0-9\-]+\/[a-z0-9][a-z0-9\-_.]*$/

export const GitnexusScanRequestSchema = z.discriminatedUnion('scope', [
  z.object({
    scope: z.literal('repo'),
    target: z.string().regex(RepoTargetRe).refine(
      (s) => {
        const slash = s.indexOf('/')
        if (slash < 0) return false
        const repo = s.slice(slash + 1)
        // Reject any segment that IS `.` / `..` OR contains `..` anywhere.
        // (Regex's leading [a-z0-9] already prevents leading-dot and leading-hyphen;
        //  this refinement closes the nested-dotdot case.)
        if (repo === '.' || repo === '..') return false
        if (repo.includes('..')) return false
        return true
      },
      { message: 'repo target cannot contain `..` or be `.`/`..`' },
    ),
  }).strict(),
  z.object({ scope: z.literal('family'), target: z.enum(['agenticapps','factiv','neuroflash']) }).strict(),
])
```

```ts
export function deterministicRepoRoot(repoId: string): string | null {
  const slash = repoId.indexOf('/')
  if (slash < 1 || slash === repoId.length - 1) return null
  const family = repoId.slice(0, slash)
  const repo = repoId.slice(slash + 1)
  const knownFamilies = ['agenticapps', 'factiv', 'neuroflash'] as const
  if (!(knownFamilies as readonly string[]).includes(family)) return null
  if (repo.includes('/') || repo.includes(sep)) return null
  // Defence-in-depth: reject `.`/`..` here too (wire schema already does, but the
  // helper may be called from non-route paths in future).
  if (repo === '.' || repo === '..' || repo.includes('..')) return null

  const home = homedir()
  const familyPrefix = `${home}${sep}Sourcecode${sep}${family}${sep}`
  const root = `${familyPrefix}${repo}`
  if (!existsSync(root)) return null
  try {
    if (!statSync(root).isDirectory()) return null
  } catch {
    return null
  }

  // CRITICAL #2 — Symlink escape defence. Realpath the candidate and refuse if
  // it resolves OUTSIDE the family root. Mirrors the assertSnapshotDirInDaemonHome
  // idiom in server/boot.ts:98-116 and the matching check in coverageScan.ts.
  let real: string
  try {
    real = realpathSync.native(root)
  } catch {
    return null
  }
  // Accept iff real === root OR real lives under familyPrefix. We require
  // start-with-familyPrefix because anywhere under the same family is fine
  // (alias symlinks within the family root); escapes outside are not.
  if (real !== root && !real.startsWith(familyPrefix)) return null
  return root
}
```

```ts
/**
 * D-13-EXT-09 — FS-aligned family repo derivation.
 *
 * Reads ~/Sourcecode/{family}/ directly and accepts each subdirectory that
 * passes deterministicRepoRoot() (which enforces family allow-list, directory
 * existence, AND realpath stays under the family root after Task 2's hardening).
 *
 * Used by startFamilyScan instead of the older deriveRepos(registry.entries, ...)
 * so a family scan covers every repo visible in the Coverage matrix, not only
 * the subset that happens to be in the dashboard registry.
 *
 * The older deriveRepos() helper is kept (unused) pending post-merge cleanup.

export function deriveFamilyReposFromFs(
  familyId: KnownFamily,
): Array<{ repo: string; root: string }> {
  const familyRoot = `${homedir()}${sep}Sourcecode${sep}${familyId}`
  if (!existsSync(familyRoot)) return []
  let entries: string[]
  try {
    entries = readdirSync(familyRoot)
  } catch {
    return []
  }
  const result: Array<{ repo: string; root: string }> = []
  for (const name of entries) {
    // Skip hidden dirs (.git, .DS_Store, etc.)
    if (name.startsWith('.')) continue
    const resolved = deterministicRepoRoot(`${familyId}/${name}`)
    if (resolved !== null) {
      result.push({ repo: name, root: resolved })
    }
  }
  // Preserve D-13-04 ordering — alphabetical by repo name.
  result.sort((a, b) => a.repo.localeCompare(b.repo))
  return result
}
```
