/**
 * D-13-02 short-poll contract: synchronous register + fire-and-forget body.
 * Returns within microseconds (just enough to validate the family + register the job).
 * The orchestration loop runs in the background via `void startFamilyScanBody(...)`.
 *
 * MIRRORS the per-repo startScan pattern at gitnexusScan.ts:154-186 verbatim.
 *
 * NOTE: 'BINARY_NOT_FOUND' is no longer in the return union — the upfront resolve
 * was synchronous-impossible (the resolve was an async call buried in the body).
 * T-13-06-05 documents the UX regression and rationale.
 */
export function startFamilyScan(
  familyScanId: string,
  familyId: KnownFamily,
  registry: { entries: ... },
  opts: { registryFile?: string } = {},
): { ok: true } | { ok: false; code: 'FAMILY_HAS_NO_REPOS' } {
  const repos = deriveRepos(registry.entries, familyId).sort(
    (a, b) => a.repo.localeCompare(b.repo),
  )
  if (repos.length === 0) return { ok: false, code: 'FAMILY_HAS_NO_REPOS' }
  registerFamilyJob(familyScanId, familyId, repos)
  // Fire-and-forget — the body runs the for-of loop + finalization.
  // Wrapped in `void` so unhandled-promise warnings are explicit.
  void startFamilyScanBody(familyScanId, familyId, repos, opts).catch((err) => {
    // Defensive: a thrown error inside the body should NOT leave the family
    // job stuck in 'running'. Finalize as done with all-failed counts.
    console.error('[gitnexusFamilyScan] unhandled body error', err)
    updateFamilyJob(familyScanId, (s) => ({
      ...s,
      state: 'done' as const,
      completedAt: new Date().toISOString(),
      currentRepoId: null,
      currentScanId: null,
    }))
    scheduleFamilyEviction(familyScanId)
  })
  return { ok: true }
}
