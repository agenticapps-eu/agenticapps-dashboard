```typescript
// packages/spa/src/lib/queries/gitnexusScan.ts (NEW)
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
import {
  GitnexusScanRequestSchema,
  GitnexusScanResponseSchema,
  GitnexusScanProgressSchema,
  type GitnexusScanRequest,
  type GitnexusScanErrorCode,
} from '@agenticapps/dashboard-shared'
import { apiFetch } from '../api'

export function useGitnexusScan() {
  return useMutation({
    mutationFn: async (body: GitnexusScanRequest) => {
      const r = await apiFetch('/api/gitnexus/scan', GitnexusScanResponseSchema, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(GitnexusScanRequestSchema.parse(body)),
      })
      if (!r.ok) throw new Error(`schema_drift:${r.drift.path}`)
      if (!r.data.ok) {
        const err = new Error(r.data.error) as Error & { code: GitnexusScanErrorCode }
        err.code = r.data.error
        throw err
      }
      return r.data    // { ok: true, scanId }
    },
    // Coverage invalidation happens at the consumer (after polling reaches 'done'), NOT here.
  })
}

export function useGitnexusScanProgress(scanId: string | null) {
  const qc = useQueryClient()
  const query = useQuery({
    queryKey: ['gitnexusScan', scanId],
    queryFn: async () => {
      const r = await apiFetch(`/api/gitnexus/scan/${scanId}`, GitnexusScanProgressSchema)
      if (!r.ok) throw new Error(`schema_drift:${r.drift.path}`)
      if (!r.data.ok) {
        const err = new Error(r.data.error) as Error & { code: GitnexusScanErrorCode }
        err.code = r.data.error
        throw err
      }
      return r.data.job
    },
    enabled: scanId !== null,
    refetchInterval: (q) => {
      const job = q.state.data
      if (!job) return 1500
      return job.state === 'running' ? 1500 : false
    },
    staleTime: 0,
    gcTime: 60_000,
    refetchOnWindowFocus: false,
  })
  // NOTE: invalidation is triggered by the CONSUMER via useEffect on terminal state,
  // not inside this hook. See ScanPill / CoverageFamilySection usage below.
  return query
}

export function scanErrorCodeToMessage(code: GitnexusScanErrorCode): string {
  switch (code) {
    case 'BINARY_NOT_FOUND':     return 'GitNexus is not installed — install and try again'
    case 'REPO_NOT_REGISTERED':  return 'Repo not found in registry — re-register and try again'
    case 'FAMILY_HAS_NO_REPOS':  return 'No repos in this family'
    case 'SCAN_IN_FLIGHT':       return 'A scan is already running for this repo'
    case 'BIND_REFUSED':         return 'Scan disabled — connect from the host device'
    case 'RATE_LIMITED':         return 'Too many requests — try again in a few seconds'
    case 'SCAN_NOT_FOUND':       return 'Scan was interrupted'
    case 'SCAN_FAILED':          return 'Scan failed — see daemon logs'
    case 'SCAN_TIMEOUT':         return 'Scan timed out after 5 minutes'
    case 'INVALID_REQUEST':      return 'Invalid request — please reload'
    case 'INTERNAL_ERROR':       return 'Scan failed — see daemon logs'
  }
}
```

```tsx
// packages/spa/src/components/panels/coverage/ScanPill.tsx (NEW)
// Per-row + per-family Scan affordance. Composes hooks + Toast + Tooltip.
//
// Props:
//   scope: 'repo' | 'family'
//   target: string                       // repoId (family/repo) OR familyId
//   canScan: boolean                      // from health response
//   installed: boolean                    // from health response
//
// Returns null when installed === false (parent renders InstallGitNexusButton instead).
//
// CRITICAL: All React hooks must be called UNCONDITIONALLY at the top of the function,
// BEFORE the `if (!installed) return null` early exit (Rules of Hooks).

import { Sparkles, Loader2, Play } from 'lucide-react'
import { useState, useEffect } from 'react'
import { useQueryClient } from '@tanstack/react-query'
import { useGitnexusScan, useGitnexusScanProgress, scanErrorCodeToMessage } from '../../../lib/queries/gitnexusScan'
import { useToast } from '../../ui/Toast'
import { Tooltip } from '../../ui/Tooltip'

interface ScanPillProps {
  scope: 'repo' | 'family'
  target: string
  canScan: boolean
  installed: boolean
}

export function ScanPill({ scope, target, canScan, installed }: ScanPillProps) {
  // All hooks first — Rules of Hooks
  const qc = useQueryClient()
  const toast = useToast()
  const [scanId, setScanId] = useState<string | null>(null)
  const scan = useGitnexusScan()
  const progress = useGitnexusScanProgress(scanId)

  useEffect(() => {
    if (!progress.data) return
    if (progress.data.state === 'running') return
    // terminal — invalidate cache + toast + reset
    qc.invalidateQueries({ queryKey: ['coverage'] })
    qc.invalidateQueries({ queryKey: ['conformance'] })
    if (progress.data.kind === 'repo') {
      if (progress.data.state === 'done') {
        toast.show({ variant: 'success', message: `Indexed ${target}` })
      } else {
        const code = progress.data.error?.code ?? 'INTERNAL_ERROR'
        toast.show({ variant: 'error', message: `Indexing failed: ${scanErrorCodeToMessage(code)}` })
      }
    } else {
      // family
      const { completed, failed, total } = progress.data
      if (failed === 0) {
        toast.show({ variant: 'success', message: `Scanned ${completed} repos in ${target}` })
      } else {
        toast.show({ variant: 'error', message: `${completed}/${total} scanned, ${failed} failed — retry failed?` })
      }
    }
    setScanId(null)
  }, [progress.data?.state])

  // Now safe to return null after all hooks ran
  if (!installed) return null

  if (!canScan) {
    return (
      <Tooltip content="Connect from the host device to scan">
        <button type="button" disabled aria-disabled="true" className="opacity-50 cursor-not-allowed inline-flex items-center gap-1 text-xs">
          <Play size={12} aria-hidden /> Scan
        </button>
      </Tooltip>
    )
  }

  const isPending = scanId !== null && (progress.data?.state === 'running' || progress.isFetching)
  if (isPending) {
    return (
      <span className="inline-flex items-center gap-1 text-xs text-text-secondary" aria-live="polite">
        <Loader2 size={12} className="animate-spin" aria-hidden /> Scanning…
      </span>
    )
  }

  return (
    <button
      type="button"
      onClick={async () => {
        try {
          const r = await scan.mutateAsync({ scope, target })
          setScanId(r.scanId)
        } catch (err) {
          const code = (err as { code?: string }).code ?? 'INTERNAL_ERROR'
          toast.show({ variant: 'error', message: `Indexing failed: ${scanErrorCodeToMessage(code as never)}` })
        }
      }}
      className="inline-flex items-center gap-1 text-xs rounded-pill bg-accent-soft px-2 py-0.5 hover:bg-accent-softer focus:outline focus:outline-2 focus:outline-accent"
    >
      <Sparkles size={12} aria-hidden /> Scan
    </button>
  )
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-03-01 | Information Disclosure | Error messages displayed to the user via toast | mitigate | `scanErrorCodeToMessage(code)` maps the 11 error codes from a fixed enum to user-friendly strings. The raw `error.message` from the daemon is NEVER passed through to the toast — only `error.code`. This prevents accidental disclosure of file paths, stderr, or system internals via the toast surface. |
| T-13-03-02 | Tampering | XSS via toast message containing user-controlled data | mitigate | React JSX escapes all interpolated strings by default — `target` (the repoId/familyId) is interpolated via `${target}` in JSX, not via any raw-HTML rendering API. Even if the daemon returned a malicious string, React would render it as text. ScanPill does NOT use any escape-hatch HTML rendering API. |
| T-13-03-03 | Spoofing | User clicks Scan when they think it's already running (double-trigger) | mitigate | Per-row in-flight handling: `setScanId` is only called on successful mutateAsync resolution. The button is replaced with the "Scanning…" pill while `scanId !== null && progress.data?.state === 'running'` — the button is no longer clickable. Daemon-side per-repo lock (Plan 13-02) is the authoritative guard if the user manages to click twice anyway: returns 409 SCAN_IN_FLIGHT which toasts "A scan is already running for this repo". |
| T-13-03-04 | Denial of Service | User repeatedly clicks Scan on many rows | accept (mitigated upstream) | Rate limiter on the daemon (10/10s per token-hash, Plan 13-02 T-13-02-03) bounds the request rate. UI does not add a separate client-side cap — the daemon's 429 RATE_LIMITED with friendly toast is the user feedback. |
