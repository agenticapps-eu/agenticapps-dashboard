```typescript
import { useQuery, type UseQueryResult } from '@tanstack/react-query'

import {
  CoverageHistoryResponseSchema,
  type CoverageHistoryResponse,
} from '@agenticapps/dashboard-shared'

import { apiFetch } from './api.js'

const ONE_HOUR_MS = 60 * 60 * 1000

export type CoverageCell = 'claudeMd' | 'gitNexus' | 'wiki' | 'workflowVersion'

export interface UseCoverageHistoryOptions {
  enabled?: boolean
}

/**
 * useCoverageHistory — wraps GET /api/coverage/history?repoId=&cell=.
 *
 * staleTime is 1 hour to mirror the daemon-side `coverageHistoryCache` TTL
 * (Plan 02). The matrix endpoint changes once-per-day at most (daily cron),
 * so a 1h client cache is comfortably within the data freshness budget.
 *
 * Per-(repoId, cell) tuple gets its own cache entry — the queryKey discriminates
 * so the SPA can call this hook from many cells without cross-bleed.
 */
export function useCoverageHistory(
  repoId: string,
  cell: CoverageCell,
  opts: UseCoverageHistoryOptions = {},
): UseQueryResult<CoverageHistoryResponse, Error> {
  return useQuery<CoverageHistoryResponse, Error>({
    queryKey: ['coverageHistory', repoId, cell],
    queryFn: () =>
      apiFetch<CoverageHistoryResponse>({
        path: `/api/coverage/history?repoId=${encodeURIComponent(repoId)}&cell=${cell}`,
        schema: CoverageHistoryResponseSchema,
      }),
    staleTime: ONE_HOUR_MS,
    enabled: opts.enabled ?? true,
  })
}
```

```typescript
import type { ReactElement } from 'react'

export interface CoverageDriftBadgeProps {
  direction: 'up' | 'down'
  daysSince: number
}

/**
 * Inline drift indicator on a CoverageCell — text-only ▲Nd / ▼Nd badge.
 *
 * D-11-03 — Phase 11 chose inline text over a sparkline / SVG to preserve the
 * matrix's calm aesthetic post Phase 6 / 10.5 polish.
 *
 * Tokens:
 *   - text-status-success (▲ — improvement, e.g., missing → fresh)
 *   - text-status-error   (▼ — regression, e.g., fresh → stale)
 * Both tokens already exist in tokens.css; no new hex literals so the
 * tokenSourceOfTruth.test.ts invariant continues to pass.
 *
 * Name choice (D-11-03 footnote): NOT `InlineDrift` — that's the Phase 6
 * schema-drift panel at packages/spa/src/components/panels/InlineDrift.tsx.
 */
export function CoverageDriftBadge({
  direction,
  daysSince,
}: CoverageDriftBadgeProps): ReactElement {
  const arrow = direction === 'up' ? '▲' : '▼'
  const colorClass = direction === 'up' ? 'text-status-success' : 'text-status-error'
  const verb = direction === 'up' ? 'Improved' : 'Regressed'
  const dayWord = daysSince === 1 ? 'day' : 'days'
  return (
    <span
      className={`text-xs font-semibold ${colorClass}`}
      aria-label={`${verb} ${daysSince} ${dayWord} ago`}
    >
      {arrow}
      {daysSince}d
    </span>
  )
}
```
