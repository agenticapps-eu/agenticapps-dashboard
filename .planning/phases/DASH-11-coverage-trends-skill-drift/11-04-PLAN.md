```typescript
import { useQuery, type UseQueryResult } from '@tanstack/react-query'

import {
  CoverageHistoryResponseSchema,
  type CoverageHistoryResponse,
} from '@agenticapps/dashboard-shared'

import { apiFetch } from './api.js'

const ONE_HOUR_MS = 60 * 60 * 1000

export interface UseCoverageHistoryOptions {
  enabled?: boolean
}

/**
 * useCoverageHistory(repoId) — wraps GET /api/coverage/history?repoId=
 *
 * PD-11-02 (REVIEWS.md action item 1, Option C):
 *   - One request returns drift for ALL FOUR cells of the named repo.
 *   - Drops the `cell` parameter from the hook signature.
 *   - Consumed by CoverageRow.tsx (the single owner), which fans the four
 *     cell drifts out as props to its four CoverageCell children.
 *
 * Performance budget (REVIEWS.md action item 2): ≤ 1 fetch per (repoId, mount)
 * — TanStack dedup on the queryKey ['coverageHistory', repoId] enforces this
 * structurally.
 *
 * staleTime is 1 hour to mirror the daemon-side `coverageHistoryCache` TTL.
 * History changes once-per-day at most (daily cron), so a 1h client cache
 * is comfortably within the data freshness budget.
 */
export function useCoverageHistory(
  repoId: string,
  opts: UseCoverageHistoryOptions = {},
): UseQueryResult<CoverageHistoryResponse, Error> {
  return useQuery<CoverageHistoryResponse, Error>({
    queryKey: ['coverageHistory', repoId],
    queryFn: async () => {
      const result = await apiFetch(
        `/api/coverage/history?repoId=${encodeURIComponent(repoId)}`,
        CoverageHistoryResponseSchema,
      )
      if (!result.ok) {
        throw new Error(
          `schema_drift: ${result.drift.path} expected ${result.drift.expected} got ${result.drift.got}`,
        )
      }
      return result.data
    },
    staleTime: ONE_HOUR_MS,
    enabled: opts.enabled ?? true,
  })
}
```

export interface CoverageDriftBadgeProps {
  direction: 'up' | 'down'
  daysSince: number
}

/**
 * Inline drift indicator on a CoverageCell — text-only ▲Nd / ▼Nd badge.
 *

export function CoverageDriftBadge({
  direction,
  daysSince,
}: CoverageDriftBadgeProps): ReactElement {
  const arrow = direction === 'up' ? '▲' : '▼'

    >
      {arrow}
      {daysSince}d
    </span>
  )
}
