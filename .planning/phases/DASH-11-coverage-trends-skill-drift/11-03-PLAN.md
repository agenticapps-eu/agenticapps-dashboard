```typescript
// packages/agent/src/lib/skillDriftCache.ts
import type { SkillDriftResponse } from '@agenticapps/dashboard-shared'

const TTL_MS = 30 * 1000 // 30s — matches Phase 10 coverage cache cadence

interface Entry {
  value: SkillDriftResponse
  expiresAt: number
}

let entry: Entry | null = null

export function getSkillDriftCached(now: number = Date.now()): SkillDriftResponse | undefined {
  if (!entry) return undefined
  if (now >= entry.expiresAt) {
    entry = null
    return undefined
  }
  return entry.value
}

export function setSkillDriftCached(value: SkillDriftResponse, now: number = Date.now()): void {
  entry = { value, expiresAt: now + TTL_MS }
}

export function clearSkillDriftCache(): void {
  entry = null
}
```

```typescript
// packages/agent/src/lib/skillDriftScan.ts
import { homedir } from 'node:os'
import { join, sep } from 'node:path'

import type {
  SkillDriftCell,
  SkillDriftResponse,
  SkillDriftRow,
} from '@agenticapps/dashboard-shared'

import { readRegistry } from './registry.js'
import { readLocalSkills } from './skillsScan.js'

export const KNOWN_FAMILIES = ['agenticapps', 'factiv', 'neuroflash'] as const
export type KnownFamily = typeof KNOWN_FAMILIES[number]
export type Family = KnownFamily | 'other'

/**
 * Derive family from a project root path by prefix-matching against
 * ~/Sourcecode/{agenticapps,factiv,neuroflash}/.
 *
 * Why path-based: live registry on this machine has client=null and tags=[]
 * for every entry (verified 2026-05-16 reading ~/.agenticapps/dashboard/registry.json).
 * Family metadata is NOT in the registry; it must come from the filesystem layout.
 *
 * Falls back to 'other' for off-family registrations (e.g., ~/Documents/foo,
 * or ~/Sourcecode/something-not-agenticapps-factiv-neuroflash).
 *
 * `homedirOverride` injection enables hermetic testing without mutating env.
 */
export function familyOf(root: string, homedirOverride?: string): Family {
  const home = homedirOverride ?? homedir()
  const sourcecode = join(home, 'Sourcecode') + sep
  if (!root.startsWith(sourcecode)) return 'other'
  const rel = root.slice(sourcecode.length).split(sep)
  const head = rel[0]
  // rel.length === 1 means the root IS the family dir itself (e.g., ~/Sourcecode/agenticapps);
  // a registered project must have a child (a repo), so this is treated as 'other'.
  if (rel.length < 2 || !head) return 'other'
  return KNOWN_FAMILIES.includes(head as KnownFamily) ? (head as KnownFamily) : 'other'
}

export async function scanSkillDrift(opts?: {
  registryFile?: string
  homedirOverride?: string
}): Promise<SkillDriftResponse> {
  const reg = readRegistry(opts?.registryFile)

  // Promise.allSettled isolation (Phase 10 AGREED-2): one project's failure
  // does NOT poison the response.
  const perProject = await Promise.allSettled(
    reg.projects.map(async (p) => ({
      projectId: p.id,
      projectName: p.name,
      family: familyOf(p.root, opts?.homedirOverride),
      skills: await readLocalSkills(p.root),
    })),
  )

  // Build projects array (preserving registry order — matrix column order)
  // and parallel error map.
  const projects: SkillDriftResponse['projects'] = []
  const skillsByProject = new Map<string, Awaited<ReturnType<typeof readLocalSkills>>>()

  for (let i = 0; i < perProject.length; i += 1) {
    const settled = perProject[i]
    const entry = reg.projects[i]
    if (settled.status === 'fulfilled') {
      projects.push({
        projectId: settled.value.projectId,
        projectName: settled.value.projectName,
        family: settled.value.family,
      })
      skillsByProject.set(settled.value.projectId, settled.value.skills)
    } else {
      projects.push({
        projectId: entry.id,
        projectName: entry.name,
        family: familyOf(entry.root, opts?.homedirOverride),
        degraded: settled.reason instanceof Error
          ? settled.reason.message
          : String(settled.reason),
      })
    }
  }

  // Collect the union of all skillIds across all successful projects.
  const allSkillIds = new Set<string>()
  for (const skills of skillsByProject.values()) {
    for (const s of skills) allSkillIds.add(s.skillId)
  }

  // Build one row per skillId; fill missing projects with absent cells.
  const rows: SkillDriftRow[] = []
  for (const skillId of Array.from(allSkillIds).sort()) {
    const byProject: Record<string, SkillDriftCell> = {}
    for (const proj of projects) {
      const skills = skillsByProject.get(proj.projectId)
      const found = skills?.find((s) => s.skillId === skillId)
      byProject[proj.projectId] = found
        ? {
            present: true,
            version: found.version ?? null,
            lastModifiedIso: found.lastModifiedIso ?? null,
          }
        : { present: false, version: null, lastModifiedIso: null }
    }
    rows.push({ skillId, byProject })
  }

  return {
    schemaVersion: 1 as const,
    generatedAtIso: new Date().toISOString(),
    projects,
    rows,
  }
}
```

```typescript
  }

  const result = await runAgentLinter(entry.root)
  const cachedAt = new Date().toISOString()
  setAgentLinterCached(entry.id, { result, cachedAt, maxMtime })
  return c.json({ ok: true, result, cachedAt })
})
```
