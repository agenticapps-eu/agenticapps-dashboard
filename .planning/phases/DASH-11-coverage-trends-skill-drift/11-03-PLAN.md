// packages/agent/src/lib/skillDriftCache.ts
import type { SkillDriftResponse } from '@agenticapps/dashboard-shared'

const TTL_MS = 30 * 1000 // 30s — matches Phase 10 coverage cache cadence

interface Entry {
  value: SkillDriftResponse
  expiresAt: number
}

let entry: Entry | null = null

export function getSkillDriftCached(now: number = Date.now()): SkillDriftResponse | undefined {
  if (!entry) return undefined
  if (now >= entry.expiresAt) {
    entry = null
    return undefined
  }
  return entry.value
}

export function setSkillDriftCached(value: SkillDriftResponse, now: number = Date.now()): void {
  entry = { value, expiresAt: now + TTL_MS }
}

export function clearSkillDriftCache(): void {
  entry = null
}
```

// packages/agent/src/lib/skillDriftScan.ts
import { homedir } from 'node:os'
import { join, sep } from 'node:path'

import type {
  SkillDriftCell,
  SkillDriftResponse,
  SkillDriftRow,
} from '@agenticapps/dashboard-shared'

import { readRegistry } from './registry.js'
import { readLocalSkills } from './skillsScan.js'

export const KNOWN_FAMILIES = ['agenticapps', 'factiv', 'neuroflash'] as const
export type KnownFamily = typeof KNOWN_FAMILIES[number]
export type Family = KnownFamily | 'other'

export function familyOf(root: string, homedirOverride?: string): Family {
  const home = homedirOverride ?? homedir()
  const sourcecode = join(home, 'Sourcecode') + sep
  if (!root.startsWith(sourcecode)) return 'other'
  const rel = root.slice(sourcecode.length).split(sep)
  const head = rel[0]
  if (rel.length < 2 || !head) return 'other'
  return KNOWN_FAMILIES.includes(head as KnownFamily) ? (head as KnownFamily) : 'other'
}

export async function scanSkillDrift(opts?: {
  registryFile?: string
  homedirOverride?: string
}): Promise<SkillDriftResponse> {
  const reg = readRegistry(opts?.registryFile)

  // Promise.allSettled isolation (Phase 10 AGREED-2): one project's failure
  // does NOT poison the response.
  const perProject = await Promise.allSettled(
    reg.projects.map(async (p) => {
      // REVIEWS.md action item 6: readLocalSkills returns { scope: 'local'; skills: SkillEntry[] }.
      // Destructure .skills explicitly — do NOT treat the return as a raw array.
      const { skills } = await readLocalSkills(p.root)
      return {
        projectId: p.id,
        projectName: p.name,
        family: familyOf(p.root, opts?.homedirOverride),
        skills,
      }
    }),
  )

  const projects: SkillDriftResponse['projects'] = []
  const skillsByProject = new Map<string, Awaited<ReturnType<typeof readLocalSkills>>['skills']>()

  for (let i = 0; i < perProject.length; i += 1) {
    const settled = perProject[i]
    const entry = reg.projects[i]
    if (settled.status === 'fulfilled') {
      projects.push({
        projectId: settled.value.projectId,
        projectName: settled.value.projectName,
        family: settled.value.family,
      })
      skillsByProject.set(settled.value.projectId, settled.value.skills)
    } else {
      projects.push({
        projectId: entry.id,
        projectName: entry.name,
        family: familyOf(entry.root, opts?.homedirOverride),
        degraded: settled.reason instanceof Error
          ? settled.reason.message
          : String(settled.reason),
      })
    }
  }

  // NOTE: read the actual SkillEntry field names from skillsScan.ts and adapt
  // the field accessors below (`s.skillId` / `s.version` / `s.lastModifiedIso`)
  // to whatever the real type uses. The destructuring of `.skills` is fixed;
  // the inner field names follow the live source.
  const allSkillIds = new Set<string>()
  for (const skills of skillsByProject.values()) {
    for (const s of skills) allSkillIds.add(s.skillId ?? s.name ?? '<unknown>')
  }

  const rows: SkillDriftRow[] = []
  for (const skillId of Array.from(allSkillIds).sort()) {
    const byProject: Record<string, SkillDriftCell> = {}
    for (const proj of projects) {
      const skills = skillsByProject.get(proj.projectId)
      const found = skills?.find((s) => (s.skillId ?? s.name) === skillId)
      byProject[proj.projectId] = found
        ? {
            present: true,
            version: found.version ?? null,
            lastModifiedIso: found.lastModifiedIso ?? null,
          }
        : { present: false, version: null, lastModifiedIso: null }
    }
    rows.push({ skillId, byProject })
  }

  return {
    schemaVersion: 1 as const,
    generatedAtIso: new Date().toISOString(),
    projects,
    rows,
  }
}
```

// packages/agent/src/routes/skillDrift.ts
import { Hono } from 'hono'
import { z } from 'zod'

import {
  AgentLinterResponseSchema,         // REVIEWS action item 10 — shared schema reused
  SkillDriftResponseSchema,
} from '@agenticapps/dashboard-shared'

import {
  computeMaxMtime,
  getAgentLinterCached,
  setAgentLinterCached,
} from '../lib/agentLinterCache.js'
import { runAgentLinter } from '../lib/agentLinterRunner.js'
import { readRegistry } from '../lib/registry.js'
import {
  clearSkillDriftCache,
  getSkillDriftCached,
  setSkillDriftCached,
} from '../lib/skillDriftCache.js'
import { scanSkillDrift } from '../lib/skillDriftScan.js'
import { outbound } from '../server/middleware/errors.js'
import type { Env } from '../server/app.js'

export const skillDriftRoute = new Hono<Env>()

// --- GET /skills/drift -------------------------------------------------------
skillDriftRoute.get('/skills/drift', async (c) => {
  const cached = getSkillDriftCached()
  const response = cached ?? (await scanSkillDrift())
  if (!cached) setSkillDriftCached(response)
  return outbound(c, SkillDriftResponseSchema.parse.bind(SkillDriftResponseSchema), response)
})

// --- POST /skills/drift/agentlinter ------------------------------------------
// D-11-14: single-project-per-request enforced structurally.
// `.strict()` rejects extra fields so a typo or smuggled `projectIds: [...]`
// surfaces as 400, not silent acceptance.
const AgentLinterDriftRequestSchema = z
  .object({
    projectId: z.string().min(1),
  })
  .strict()

skillDriftRoute.post('/skills/drift/agentlinter', async (c) => {
  let body: { projectId: string }
  try {
    body = AgentLinterDriftRequestSchema.parse(await c.req.json())
  } catch {
    return c.json({ ok: false, error: 'invalid_request_body' }, 400)
  }

  // Registry lookup — fails closed (404) when project is unknown.
  const reg = readRegistry()
  const entry = reg.projects.find((p) => p.id === body.projectId)
  if (!entry) {
    return c.json({ ok: false, error: 'project_not_found' }, 404)
  }

  // Reuse Phase 5 cache + runner UNCHANGED (D-11-14).
  const maxMtime = await computeMaxMtime(entry.root)
  const cached = getAgentLinterCached(entry.id, maxMtime)
  if (cached) {
    // REVIEWS action item 10 — wrap via shared AgentLinterResponseSchema (same as Phase 5 GET).
    return outbound(
      c,
      AgentLinterResponseSchema.parse.bind(AgentLinterResponseSchema),
      cached.result,
    )
  }

  const result = await runAgentLinter(entry.root)
  const cachedAt = new Date().toISOString()
  setAgentLinterCached(entry.id, { result, cachedAt, maxMtime })
  return outbound(
    c,
    AgentLinterResponseSchema.parse.bind(AgentLinterResponseSchema),
    result,
  )
})

// Exposed for test isolation
export const __test__ = { clearSkillDriftCache }
