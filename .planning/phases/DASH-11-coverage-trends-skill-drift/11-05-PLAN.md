// packages/spa/src/lib/skillDriftQueries.ts
import {
  useMutation,
  useQuery,
  useQueryClient,
  type UseMutationResult,
  type UseQueryResult,
} from '@tanstack/react-query'
import { z } from 'zod'

import {
  SkillDriftResponseSchema,
  type SkillDriftResponse,
} from '@agenticapps/dashboard-shared'

import { apiFetch } from './api.js'

const THIRTY_SECONDS_MS = 30 * 1000

// The mutation response shape mirrors Phase 5's AgentLinter GET response.
// We parse loosely on the SPA side because the AgentLinter result shape is
// already validated by the daemon's outbound() wrapper using the existing
// AgentLinterResponseSchema. SPA-side just confirms `ok: true`.
const AgentLinterDriftResponseSchema = z.object({
  ok: z.literal(true),
  result: z.unknown(), // delegated to Phase 5 schema enforcement
  cachedAt: z.string(),
})
type AgentLinterDriftResponse = z.infer<typeof AgentLinterDriftResponseSchema>

// apiFetch signature (verified by reading packages/spa/src/lib/api.ts):
//   apiFetch<S>(path: string, schema: S, init: RequestInit = {}): Promise<ParseOrDrift<z.infer<S>>>
//
// It already supports POST natively via init.method + init.body; no extension needed.
// It also already handles bearer auth, 401 repair, and parseOrDrift on the response.
// One quirk: apiFetch returns ParseOrDrift<T>, NOT T directly — callers unwrap.

export function useSkillDrift(): UseQueryResult<SkillDriftResponse, Error> {
  return useQuery<SkillDriftResponse, Error>({
    queryKey: ['skillDrift'],
    queryFn: async () => {
      const result = await apiFetch('/api/skills/drift', SkillDriftResponseSchema)
      if (!result.ok) {
        throw new Error(`schema_drift: ${result.drift.path} expected ${result.drift.expected} got ${result.drift.got}`)
      }
      return result.data
    },
    staleTime: THIRTY_SECONDS_MS,
    refetchInterval: 5_000, // mirrors Phase 10 cadence
  })
}

export interface UseAgentLinterDriftVariables {
  projectId: string
}

export function useAgentLinterDrift(): UseMutationResult<
  AgentLinterDriftResponse,
  Error,
  UseAgentLinterDriftVariables
> {
  const queryClient = useQueryClient()
  return useMutation<AgentLinterDriftResponse, Error, UseAgentLinterDriftVariables>({
    mutationFn: async ({ projectId }) => {
      const result = await apiFetch(
        '/api/skills/drift/agentlinter',
        AgentLinterDriftResponseSchema,
        {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ projectId }), // EXACTLY {projectId} — D-11-14 single-project enforcement
        },
      )
      if (!result.ok) {
        throw new Error(`schema_drift: ${result.drift.path} expected ${result.drift.expected} got ${result.drift.got}`)
      }
      return result.data
    },
    onSuccess: () => {
      // Trigger matrix refetch so the freshly-linted cell reflects new state.
      queryClient.invalidateQueries({ queryKey: ['skillDrift'] })
    },
  })
}
```

NOTE on apiFetch: verified at planning time that `apiFetch(path, schema, init)` already accepts `RequestInit` (including `method: 'POST'`, `body: string`, `headers`). NO api.ts extension needed — Plan 05 uses the existing surface unchanged.

// packages/spa/src/components/panels/skill-drift/SkillDriftCell.tsx
import type { ReactElement } from 'react'
import type { SkillDriftCell as SkillDriftCellData } from '@agenticapps/dashboard-shared'

export interface SkillDriftCellProps {
  cell: SkillDriftCellData
  skillId: string
  projectName: string
}

export function SkillDriftCell({ cell, skillId, projectName }: SkillDriftCellProps): ReactElement {
  if (!cell.present) {
    return (
      <span
        className="text-text-tertiary text-xs"
        aria-label={`${skillId} absent in ${projectName}`}
      >
        ✕
      </span>
    )
  }
  return (
    <span
      className="text-text-secondary text-xs"
      aria-label={
        cell.version
          ? `${skillId} present in ${projectName}, version ${cell.version}`
          : `${skillId} present in ${projectName}, version unknown`
      }
    >
      ✓ {cell.version ?? <span className="text-text-tertiary">version unknown</span>}
    </span>
  )
}
```
