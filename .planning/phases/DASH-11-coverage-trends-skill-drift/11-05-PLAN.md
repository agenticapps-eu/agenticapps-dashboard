```typescript
// packages/spa/src/lib/skillDriftQueries.ts
import {
  useMutation,
  useQuery,
  useQueryClient,
  type UseMutationResult,
  type UseQueryResult,
} from '@tanstack/react-query'

import {
  AgentLinterResponseSchema,   // REVIEWS action item 10 — SHARED schema
  SkillDriftResponseSchema,
  type AgentLinterResponse,
  type SkillDriftResponse,
} from '@agenticapps/dashboard-shared'

import { apiFetch } from './api.js'

const THIRTY_SECONDS_MS = 30 * 1000

export type SkillDriftScope = 'family' | 'cross'

export interface UseSkillDriftOptions {
  scope?: SkillDriftScope
}

export function useSkillDrift(
  opts: UseSkillDriftOptions = {},
): UseQueryResult<SkillDriftResponse, Error> {
  const scope = opts.scope ?? 'family'
  return useQuery<SkillDriftResponse, Error>({
    queryKey: ['skillDrift', scope],
    queryFn: async () => {
      const result = await apiFetch('/api/skills/drift', SkillDriftResponseSchema)
      if (!result.ok) {
        throw new Error(
          `schema_drift: ${result.drift.path} expected ${result.drift.expected} got ${result.drift.got}`,
        )
      }
      return result.data
    },
    staleTime: THIRTY_SECONDS_MS,
    refetchInterval: 5_000,
  })
}

export interface UseAgentLinterDriftVariables {
  projectId: string
}

export function useAgentLinterDrift(): UseMutationResult<
  AgentLinterResponse,
  Error,
  UseAgentLinterDriftVariables
> {
  const queryClient = useQueryClient()
  return useMutation<AgentLinterResponse, Error, UseAgentLinterDriftVariables>({
    mutationFn: async ({ projectId }) => {
      const result = await apiFetch(
        '/api/skills/drift/agentlinter',
        AgentLinterResponseSchema,                // SHARED schema — REVIEWS action item 10
        {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ projectId }),    // D-11-14 — single field only
        },
      )
      if (!result.ok) {
        throw new Error(
          `schema_drift: ${result.drift.path} expected ${result.drift.expected} got ${result.drift.got}`,
        )
      }
      return result.data
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['skillDrift'] })
    },
  })
}
```

```typescript
// packages/spa/src/components/panels/skill-drift/SkillDriftMatrix.tsx
import type { SkillDriftResponse } from '@agenticapps/dashboard-shared'
import type { SkillDriftScope } from '../../../lib/skillDriftQueries.js'

const FAMILIES = ['agenticapps', 'factiv', 'neuroflash', 'other'] as const

export interface SkillDriftMatrixProps {
  data: SkillDriftResponse
  scope: SkillDriftScope
}

export function SkillDriftMatrix({ data, scope }: SkillDriftMatrixProps): JSX.Element {
  if (data.rows.length === 0) {
    return <EmptyState title="No skills detected" />
  }
  if (scope === 'cross') {
    const sortedProjects = [...data.projects].sort((a, b) =>
      a.projectId < b.projectId ? -1 : a.projectId > b.projectId ? 1 : 0,
    )
    return <SkillDriftFlatBlock rows={data.rows} projects={sortedProjects} />
  }
  // scope === 'family' — render four family sections
  return (
    <div className="flex flex-col gap-4">
      {FAMILIES.map((family) => {
        const familyProjects = data.projects.filter((p) => p.family === family)
        if (familyProjects.length === 0) return null
        return (
```
