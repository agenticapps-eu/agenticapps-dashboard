```typescript
import { existsSync, readdirSync, unlinkSync } from 'node:fs'
import { join } from 'node:path'

import { isSnapshotFilename, RETENTION_DAYS } from './snapshotPaths.js'

export interface PruneResult {
  pruned: number
  kept: number
}

/**
 * Drop snapshot files whose ISO-date filename is older than `now - retentionDays`.
 * Lazy: invoked from the writer immediately before each tick — no second scheduler.
 * Pure: derives cutoff from filename, does NOT stat (filename IS the date).
 *
 * Safety:
 *  - No-op when dir is absent (first-run case before writer ever ran).
 *  - Ignores filenames that do NOT match `isSnapshotFilename` regex.
 *  - Uses unlinkSync (synchronous): pruner runs once per day, ≤ 14 file
 *    operations per tick — measured cost ≪ 1ms.
 */
export function pruneSnapshotsOlderThan(
  dir: string,
  retentionDays: number = RETENTION_DAYS,
  now: Date = new Date(),
): PruneResult {
  if (!existsSync(dir)) return { pruned: 0, kept: 0 }

  // Cutoff: files whose date is STRICTLY older than (now - retentionDays days)
  // get pruned. Equal-to-cutoff (exactly retentionDays old) is kept.
  const cutoff = new Date(now.getTime() - retentionDays * 24 * 60 * 60 * 1000)
  const cutoffIso = cutoff.toISOString().slice(0, 10)

  let pruned = 0
  let kept = 0
  for (const name of readdirSync(dir)) {
    if (!isSnapshotFilename(name)) continue
    const dateStr = name.slice(0, 10) // 'YYYY-MM-DD'
    if (dateStr < cutoffIso) {
      unlinkSync(join(dir, name))
      pruned += 1
    } else {
      kept += 1
    }
  }
  return { pruned, kept }
}
```

```typescript
import { appendFile, chmod, mkdir } from 'node:fs/promises'
import { existsSync } from 'node:fs'
import { join } from 'node:path'

import { scanCoverageInternal } from '../coverageScan.js'
import { pruneSnapshotsOlderThan } from './snapshotPruner.js'
import {
  isoDateFromDate,
  RETENTION_DAYS,
  resolveSnapshotDir,
} from './snapshotPaths.js'

export interface SnapshotRecord {
  ts: string
  family: string
  repo: string
  claudeMd: string
  gitNexus: string
  wiki: string
  workflowVersion: string
}

export async function writeDailySnapshot(opts?: {
  now?: Date
  dir?: string
}): Promise<{ written: number; path: string }> {
  const now = opts?.now ?? new Date()
  const dir = opts?.dir ?? resolveSnapshotDir()
  const isoDate = isoDateFromDate(now)
  const path = join(dir, `${isoDate}.ndjson`)

  // Lazy directory creation, mode 0o700 (matches Phase 6 logDir pattern).
  if (!existsSync(dir)) {
    await mkdir(dir, { recursive: true, mode: 0o700 })
  }

  // Lazy prune BEFORE write. Pitfall 1 (writer-after-prune ordering) ensures
  // the directory shrinks as days roll off the back of the 14d window.
  pruneSnapshotsOlderThan(dir, RETENTION_DAYS, now)

  const { response } = await scanCoverageInternal()
  const isFirstWriteForDay = !existsSync(path)

  // Build all lines in memory, single appendFile call (Pitfall 3 defence —
  // one syscall guarantees the lines do not interleave with anything).
  const lines = response.rows.map((row): SnapshotRecord => ({
    ts: now.toISOString(),
    family: row.family,
    repo: row.repo,
    claudeMd: row.claudeMd.state,
    gitNexus: row.gitNexus.state,
    wiki: row.wiki.state,
    workflowVersion: row.workflowVersion.state,
  }))

  const body = lines.map((r) => JSON.stringify(r)).join('\n') + '\n'
  // flag: 'a' = O_APPEND; mode: 0o600 applies ONLY on creation (Pitfall 2).
  await appendFile(path, body, { flag: 'a', mode: 0o600, encoding: 'utf8' })

  // Pitfall 2 defence: explicit chmod after first creation. POSIX open(2)
  // honours `mode` only when the file is created; subsequent appends do NOT
  // re-chmod. We chmod once-per-creation (cheap; once per day) so mode stays
  // 0o600 even if umask was loose at create-time.
  if (isFirstWriteForDay) {
    await chmod(path, 0o600)
  }

  return { written: lines.length, path }
}
```

```typescript
import { existsSync, readFileSync, readdirSync } from 'node:fs'
import { join } from 'node:path'

import type { CoverageDriftDirection } from '@agenticapps/dashboard-shared'

import {
  isSnapshotFilename,
  isoDateFromDate,
  resolveSnapshotDir,
} from './snapshotPaths.js'

export interface DriftSummary {
  direction: CoverageDriftDirection | null
  daysSince: number | null
}

interface SnapshotLine {
  ts: string
  family: string
  repo: string
  claudeMd: string
  gitNexus: string
  wiki: string
  workflowVersion: string
}

type CellKey = 'claudeMd' | 'gitNexus' | 'wiki' | 'workflowVersion'

// 'not-applicable' is special — transitions into/out of it are NOT drift signal.
// Only fresh ↔ stale ↔ missing transitions count.
const SIGNAL_STATES = new Set(['fresh', 'stale', 'missing'])

// 'up' = improvement (toward fresh); 'down' = regression (away from fresh).
// Direction matrix:
//   missing → stale = up      missing → fresh = up      stale → fresh = up
//   fresh   → stale = down    fresh   → missing = down  stale → missing = down
function classifyTransition(prev: string, next: string): CoverageDriftDirection | null {
  if (!SIGNAL_STATES.has(prev) || !SIGNAL_STATES.has(next)) return null
  if (prev === next) return null
  const rank: Record<string, number> = { missing: 0, stale: 1, fresh: 2 }
  return rank[next] > rank[prev] ? 'up' : 'down'
}

export async function readDriftForCell(
  repoId: string,
  cell: CellKey,
  opts: { dir?: string; now?: Date } = {},
): Promise<DriftSummary> {
  const dir = opts.dir ?? resolveSnapshotDir()
  const now = opts.now ?? new Date()
  if (!existsSync(dir)) return { direction: null, daysSince: null }

  // Collect filtered & sorted filenames (ascending — oldest first).
  const files = readdirSync(dir)
    .filter(isSnapshotFilename)
    .sort()

  // For each file, in order, extract the cell state for the matching repoId.
  // We walk the resulting time-series and find the MOST-RECENT transition.
  const series: Array<{ date: string; state: string }> = []
  for (const f of files) {
    const path = join(dir, f)
    const raw = readFileSync(path, 'utf8')
    for (const line of raw.split('\n')) {
      if (!line) continue
      let rec: SnapshotLine
      try {
        rec = JSON.parse(line) as SnapshotLine
      } catch {
        continue // Pitfall 3 resilience — skip malformed line, do not throw
      }
      const recRepoId = `${rec.family}/${rec.repo}`
      if (recRepoId !== repoId) continue
      const state = rec[cell]
      series.push({ date: f.slice(0, 10), state })
    }
  }

  // Walk backwards from the most recent to find the latest transition.
  for (let i = series.length - 1; i > 0; i -= 1) {
    const direction = classifyTransition(series[i - 1].state, series[i].state)
    if (direction !== null) {
      const transitionDate = new Date(`${series[i].date}T00:00:00.000Z`)
      const daysSince = Math.max(
        0,
        Math.floor((now.getTime() - transitionDate.getTime()) / (24 * 60 * 60 * 1000)),
      )
      return { direction, daysSince }
    }
  }

  return { direction: null, daysSince: null }
}
```

```typescript
// packages/agent/src/lib/coverageHistoryCache.ts
const TTL_MS = 60 * 60 * 1000 // 1 hour

interface Entry {
  value: unknown
  expiresAt: number
}

const cache = new Map<string, Entry>()

function key(repoId: string, cell: string): string {
  return `${repoId}:${cell}`
}

export function getCoverageHistoryCached<T>(
  repoId: string,
  cell: string,
  now: number = Date.now(),
): T | undefined {
  const entry = cache.get(key(repoId, cell))
  if (!entry) return undefined
  if (now >= entry.expiresAt) {
    cache.delete(key(repoId, cell))
    return undefined
  }
  return entry.value as T
}

export function setCoverageHistoryCached<T>(
  repoId: string,
  cell: string,
  value: T,
  now: number = Date.now(),
): void {
  cache.set(key(repoId, cell), { value, expiresAt: now + TTL_MS })
}

export function clearCoverageHistoryCache(): void {
  cache.clear()
}
```

```typescript
import { existsSync } from 'node:fs'
import { join } from 'node:path'

import { agentError } from '../logging.js'
import {
  isoDateFromDate,
  resolveSnapshotDir,
} from './snapshotPaths.js'
import { writeDailySnapshot } from './snapshotWriter.js'

export interface SchedulerOptions {
  now?: () => Date
  dir?: string
  /** Hour of local time to fire (default 03 — off-peak, post-DST-safe) */
  hourLocal?: number
}

export function startSnapshotScheduler(opts: SchedulerOptions = {}): () => void {
  const now = opts.now ?? (() => new Date())
  const dir = opts.dir ?? resolveSnapshotDir()
  const hourLocal = opts.hourLocal ?? 3

  let activeTimer: NodeJS.Timeout | null = null

  function tick(): void {
    writeDailySnapshot({ now: now(), dir })
      .catch((err) => {
        agentError('snapshotScheduler', 'writeDailySnapshot failed', err)
      })
      .finally(() => scheduleNext())
  }

  function scheduleNext(): void {
    const current = now()
    const next = new Date(current)
    next.setHours(hourLocal, 0, 0, 0)
    if (next.getTime() <= current.getTime()) {
      next.setDate(next.getDate() + 1)
    }
    const msUntil = next.getTime() - current.getTime()
    activeTimer = setTimeout(tick, msUntil)
    activeTimer.unref()
  }

  // First-boot-fires-immediately IF today's snapshot file doesn't exist yet
  // (research Open Question 1 resolution).
  const todayPath = join(dir, `${isoDateFromDate(now())}.ndjson`)
  if (!existsSync(todayPath)) {
    activeTimer = setTimeout(tick, 0)
    activeTimer.unref()
  } else {
    scheduleNext()
  }

  return (): void => {
    if (activeTimer !== null) {
      clearTimeout(activeTimer)
      activeTimer = null
    }
  }
}
```
