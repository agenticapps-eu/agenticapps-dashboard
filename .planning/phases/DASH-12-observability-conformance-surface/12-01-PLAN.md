    import type { CoverageResponse, CoverageRow, CoverageFamily } from '@agenticapps/dashboard-shared'

    export interface FamilyScore {
      green: number
      amber: number
      red: number
      total: number
      score: number
    }

    function scoreRows(rows: CoverageRow[]): FamilyScore {
      let green = 0, amber = 0, red = 0, total = 0
      for (const row of rows) {
        const cells = [row.claudeMd, row.gitNexus, row.wiki, row.workflowVersion]
        for (const cell of cells) {
          if (cell.state === 'not-applicable') continue  // Pitfall 2
          total += 1
          if (cell.state === 'fresh') green += 1
          else if (cell.state === 'stale') amber += 1
          else if (cell.state === 'missing') red += 1
        }
      }
      const score = total === 0 ? 0 : Math.round((green / total) * 100)
      return { green, amber, red, total, score }
    }

    /** Exported only for unit-test discrimination of the helper. */
    export const _scoreRowsForTests = scoreRows

    /**
     * D-12-03/05/06/07 + Pitfall 2 + Pitfall 3.
     * Fleet score = Math.round((agenticapps + factiv + neuroflash) / 3) — mean-of-3 (A8 ratified).
     * Drifted repo IDs (`${family}/${repo}`) are pre-filtered from per-family denominators.
     * 'other' family rows are excluded entirely (only 3 family cards per D-12-06).
     */
    export function computeConformanceScores(
      coverage: CoverageResponse,
      driftedRepoIds: Set<string>,
    ): Record<CoverageFamily | 'fleet', FamilyScore> {
      const byFamily: Record<'agenticapps' | 'factiv' | 'neuroflash', CoverageRow[]> = {
        agenticapps: [], factiv: [], neuroflash: [],
      }
      for (const row of coverage.rows) {
        if (row.family === 'other') continue  // D-12-06
        const id = `${row.family}/${row.repo}`
        if (driftedRepoIds.has(id)) continue  // D-12-07
        byFamily[row.family].push(row)
      }

      const agenticapps = scoreRows(byFamily.agenticapps)
      const factiv = scoreRows(byFamily.factiv)
      const neuroflash = scoreRows(byFamily.neuroflash)

      // Pitfall 3 + A8: fleet = mean of 3 family SCORES, not sum-over-rows.
      const fleetScore = Math.round((agenticapps.score + factiv.score + neuroflash.score) / 3)

      const fleet: FamilyScore = {
        green: agenticapps.green + factiv.green + neuroflash.green,
        amber: agenticapps.amber + factiv.amber + neuroflash.amber,
        red: agenticapps.red + factiv.red + neuroflash.red,
        total: agenticapps.total + factiv.total + neuroflash.total,
        score: fleetScore,
      }

      // 'other' family rendered as 0/0/0/0/0 for type completeness; D-12-06 = OOS for display.
      const other: FamilyScore = { green: 0, amber: 0, red: 0, total: 0, score: 0 }

      return { agenticapps, factiv, neuroflash, other, fleet }
    }
    ```

    /**
     * snapshotFleetReader.ts — server-side 90-day fleet conformance series.
     *
     * D-12-09: 90-day x-axis window. Reads the SAME NDJSON store Phase 11
     * writes (snapshotWriter); Phase 12 is read-only consumer.
     *
     * Pitfall 2: not-applicable excluded from BOTH numerator + denominator.
     * Pitfall 3: fleet = mean of 3 family scores (A8 ratified, NOT sum-over-rows).
     *
     * Same-day collapse (mirrors snapshotReader.ts:139): last-record-wins per
     * (date, family, repo) — Map.set semantics.
     *
     * Resilience: malformed JSON lines + non-snapshot filenames are skipped
     * silently. T-11-02-08 bounded defence; never throws on garbage.
     */
    import { existsSync, readFileSync, readdirSync } from 'node:fs'
    import { join } from 'node:path'

    import type { CoverageFamily } from '@agenticapps/dashboard-shared'

    import {
      isSnapshotFilename,
      resolveSnapshotDir,
      RETENTION_DAYS,
    } from './snapshotPaths.js'

    interface SnapshotLine {
      ts: string
      family: string
      repo: string
      claudeMd: string
      gitNexus: string
      wiki: string
      workflowVersion: string
    }

    const CELL_KEYS: ReadonlyArray<keyof Pick<SnapshotLine, 'claudeMd' | 'gitNexus' | 'wiki' | 'workflowVersion'>> =
      ['claudeMd', 'gitNexus', 'wiki', 'workflowVersion']
    const FAMILY_KEYS: ReadonlyArray<Exclude<CoverageFamily, 'other'>> =
      ['agenticapps', 'factiv', 'neuroflash']
    const SIGNAL_GREEN = 'fresh'
    const SIGNAL_STATES = new Set(['fresh', 'stale', 'missing'])

    export interface DailySeriesEntry {
      date: string  // YYYY-MM-DD UTC
      fleet: number
      agenticapps: number
      factiv: number
      neuroflash: number
    }

    export interface ReadFleetSeriesOptions {
      /** Override snapshot dir for testability. Defaults to resolveSnapshotDir(). */
      dir?: string
      /** Window size; defaults to RETENTION_DAYS (90 post Wave 0). */
      windowDays?: number
      /** Wall-clock anchor for the window cutoff (defaults to new Date()). */
      now?: Date
      /** `${family}/${repo}` strings to exclude from per-family denominators. */
      driftedRepoIds?: Set<string>
    }

    /** Compute one family's score from already-deduplicated records. Mirrors conformanceScore.scoreRows. */
    function scoreFamilyRecords(records: SnapshotLine[]): number {
      let green = 0
      let total = 0
      for (const rec of records) {
        for (const cellKey of CELL_KEYS) {
          const state = rec[cellKey]
          if (!SIGNAL_STATES.has(state)) continue  // not-applicable + unknown → exclude
          total += 1
          if (state === SIGNAL_GREEN) green += 1
        }
      }
      return total === 0 ? 0 : Math.round((green / total) * 100)
    }

    export async function readDailySeriesForFleet(
      opts: ReadFleetSeriesOptions = {},
    ): Promise<DailySeriesEntry[]> {
      const dir = opts.dir ?? resolveSnapshotDir()
      const windowDays = opts.windowDays ?? RETENTION_DAYS
      const now = opts.now ?? new Date()
      const driftedRepoIds = opts.driftedRepoIds ?? new Set<string>()

      if (!existsSync(dir)) return []

      const cutoff = new Date(now.getTime() - windowDays * 24 * 60 * 60 * 1000)
      const cutoffIso = cutoff.toISOString().slice(0, 10)

      const files = readdirSync(dir)
        .filter(isSnapshotFilename)
        .filter((name) => name.slice(0, 10) >= cutoffIso)  // window defence
        .sort()

      const result: DailySeriesEntry[] = []

      for (const filename of files) {
        const date = filename.slice(0, 10)
        let raw: string
        try {
          raw = readFileSync(join(dir, filename), 'utf8')
        } catch {
          continue
        }

        // Last-record-wins per (family, repo) within this day's file.
        const byKey = new Map<string, SnapshotLine>()
        for (const line of raw.split('\n')) {
          if (!line) continue
          let rec: SnapshotLine
          try {
            rec = JSON.parse(line) as SnapshotLine
          } catch {
            continue
          }
          if (typeof rec.family !== 'string' || typeof rec.repo !== 'string') continue
          byKey.set(`${rec.family}/${rec.repo}`, rec)
        }

        // Bucket by family, dropping drifted IDs and 'other'.
        const byFamily: Record<Exclude<CoverageFamily, 'other'>, SnapshotLine[]> = {
          agenticapps: [],
          factiv: [],
          neuroflash: [],
        }
        for (const [id, rec] of byKey) {
          if (driftedRepoIds.has(id)) continue
          if (!FAMILY_KEYS.includes(rec.family as Exclude<CoverageFamily, 'other'>)) continue
          byFamily[rec.family as Exclude<CoverageFamily, 'other'>].push(rec)
        }

        const a = scoreFamilyRecords(byFamily.agenticapps)
        const f = scoreFamilyRecords(byFamily.factiv)
        const n = scoreFamilyRecords(byFamily.neuroflash)
        const fleet = Math.round((a + f + n) / 3)  // Pitfall 3

        result.push({ date, fleet, agenticapps: a, factiv: f, neuroflash: n })
      }

      return result
    }
    ```
