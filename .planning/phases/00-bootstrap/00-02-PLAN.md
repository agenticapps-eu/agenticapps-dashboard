**1. Write `packages/agent/src/cli.test.ts`** (uses `spawnSync` exclusively — no shell exec, no command-string interpolation, addresses code-review concern about command injection):

```typescript
import { spawnSync } from 'node:child_process'
import { readFileSync } from 'node:fs'
import { resolve, dirname } from 'node:path'
import { fileURLToPath } from 'node:url'

import { describe, it, expect, beforeAll } from 'vitest'

import { HealthResponseSchema } from '@agenticapps/dashboard-shared'

const __dirname = dirname(fileURLToPath(import.meta.url))
const packageRoot = resolve(__dirname, '..')
const cliBundle = resolve(packageRoot, 'dist/cli.js')

beforeAll(() => {
  // Build the bundle once before all tests so we exercise the published artifact.
  // spawnSync (not execSync) — argv array, no shell interpretation, no injection surface.
  const build = spawnSync('pnpm', ['build'], { cwd: packageRoot, stdio: 'inherit' })
  if (build.status !== 0) {
    throw new Error(`pnpm build failed with status ${build.status}`)
  }
}, 60_000)

describe('agentic-dashboard CLI (built dist/cli.js)', () => {
  it('exits 0 and prints version on --version', () => {
    const result = spawnSync('node', [cliBundle, '--version'], { encoding: 'utf8' })
    expect(result.status).toBe(0)
    expect(result.stdout).toMatch(/0\.0\.1-alpha\.0/)
  })

  it('exits 0 and prints alpha-placeholder on start', () => {
    const result = spawnSync('node', [cliBundle, 'start'], { encoding: 'utf8' })
    expect(result.status).toBe(0)
    expect(result.stdout).toContain('alpha placeholder')
    expect(result.stdout).toContain('Phase 1')
  })

  it('emits HealthResponseSchema-valid JSON on --version --json', () => {
    const result = spawnSync('node', [cliBundle, '--version', '--json'], { encoding: 'utf8' })
    expect(result.status).toBe(0)
    const parsed = JSON.parse(result.stdout)
    expect(() => HealthResponseSchema.parse(parsed)).not.toThrow()
    expect(parsed.ok).toBe(true)
    expect(parsed.version).toBe('0.0.1-alpha.0')
    expect(typeof parsed.message).toBe('string')
  })

  it('has a node shebang on the first line', () => {
    const firstLine = readFileSync(cliBundle, 'utf8').split('\n')[0]
    expect(firstLine).toBe('#!/usr/bin/env node')
  })

  it('inlined the @agenticapps/dashboard-shared import (no runtime workspace dep)', () => {
    const contents = readFileSync(cliBundle, 'utf8')
    expect(contents).not.toMatch(/from\s+['"]@agenticapps\/dashboard-shared['"]/)
  })
})
```

Security note for executor: every subprocess call uses `spawnSync(<cmd>, [<arg>, ...])` with an argv array. There is no string-concatenation of arguments, no `shell: true`, no `execSync`. This is the canonical safe pattern for Node child processes.

|----------|-------------|
| Local build → npm registry (publish, Plan 04) | tsup-bundled output lands on public registry; supply-chain risk for downstream `npx` consumers |
| User shell → `agentic-dashboard` binary | CLI is the user-facing surface; placeholder commands but published bin path matters |
| Test runner → child process | Subprocess invocations could be a command-injection surface if argv is constructed from untrusted input |
