    ```tsx
    import { useEffect, useRef, useState } from 'react'
    import { useRename, useSetTags } from '../lib/registry.js'
    import type { RegistryListItem } from '@agenticapps/dashboard-shared'

    export interface RenameDialogProps {
      isOpen: boolean
      item: RegistryListItem | null
      onClose: () => void
    }

    export function RenameDialog({ isOpen, item, onClose }: RenameDialogProps): React.JSX.Element | null {
      const dialogRef = useRef<HTMLDialogElement>(null)
      const [name, setName] = useState('')
      const rename = useRename(item?.id ?? '')

      useEffect(() => {
        if (isOpen && item) {
          setName(item.name)
          dialogRef.current?.showModal()
        } else {
          dialogRef.current?.close()
        }
      }, [isOpen, item])

      if (!item) return null

      async function handleSave() {
        if (!name.trim() || name.trim() === item!.name) { onClose(); return }
        try {
          await rename.mutateAsync({ name: name.trim() })
          onClose()
        } catch {
          /* error rendered inline; for v1 we just leave the dialog open and let the user retry */
        }
      }

      return (
        <dialog ref={dialogRef} onCancel={(e) => { e.preventDefault(); onClose() }} className="bg-[--surface] border border-[--border-strong] rounded-lg p-6 max-w-md w-full mx-4 backdrop:bg-black/60">
          <h2 className="text-xl font-semibold leading-snug text-[--text]">Rename project</h2>
          <label className="block text-sm font-semibold text-[--text] mt-4">Name</label>
          <input
            type="text" value={name} onChange={(e) => setName(e.target.value)}
            onKeyDown={(e) => e.key === 'Enter' ? handleSave() : null}
            className="w-full bg-[--surface-elevated] border border-[--border-strong] rounded-md px-3 py-2 text-sm text-[--text] mt-1"
            autoFocus
          />
          <div className="flex justify-end gap-2 mt-4">
            <button type="button" onClick={onClose} className="bg-[--surface-elevated] border border-[--border] text-sm px-3 py-2 rounded-md">Cancel</button>
            <button type="button" onClick={handleSave} disabled={rename.isPending} className="bg-[--accent] text-[--accent-fg] text-sm px-3 py-2 rounded-md hover:bg-[--accent-hover]">{rename.isPending ? 'Saving…' : 'Save'}</button>
          </div>
        </dialog>
      )
    }

    export interface EditTagsDialogProps {
      isOpen: boolean
      item: RegistryListItem | null
      existingTags: string[]  // from MultiProjectHome registry
      onClose: () => void
    }

    export function EditTagsDialog({ isOpen, item, existingTags, onClose }: EditTagsDialogProps): React.JSX.Element | null {
      const dialogRef = useRef<HTMLDialogElement>(null)
      const [tags, setTags] = useState<string[]>([])
      const [input, setInput] = useState('')
      const setTagsM = useSetTags(item?.id ?? '')

      useEffect(() => {
        if (isOpen && item) {
          setTags(item.tags)
          setInput('')
          dialogRef.current?.showModal()
        } else {
          dialogRef.current?.close()
        }
      }, [isOpen, item])

      if (!item) return null

      function toggle(tag: string) {
        setTags((prev) => prev.includes(tag) ? prev.filter(t => t !== tag) : [...prev, tag])
      }
      function addInputTag() {
        const t = input.trim()
        if (t && !tags.includes(t)) setTags([...tags, t])
        setInput('')
      }
      async function handleSave() {
        try {
          await setTagsM.mutateAsync({ tags })
          onClose()
        } catch { /* leave open */ }
      }

      const allChips = Array.from(new Set([...existingTags, ...tags]))

      return (
        <dialog ref={dialogRef} onCancel={(e) => { e.preventDefault(); onClose() }} className="bg-[--surface] border border-[--border-strong] rounded-lg p-6 max-w-md w-full mx-4 backdrop:bg-black/60">
          <h2 className="text-xl font-semibold leading-snug text-[--text]">Edit tags</h2>
          <div className="flex flex-wrap gap-2 mt-4">
            {allChips.map(t => (
              <button key={t} type="button" onClick={() => toggle(t)} aria-pressed={tags.includes(t)}
                className={tags.includes(t)
                  ? 'bg-[--accent] text-[--accent-fg] border border-[--accent] text-sm px-2 py-1 rounded'
    ```

    ```tsx
    import { useEffect, useRef, useState } from 'react'
    import { useRegistryList, filterAndSort, type SortKey } from '../lib/registry.js'
    import { HomeLayout } from './HomeLayout.js'
    import { HomeToolbar } from './HomeToolbar.js'
    import { ProjectCard } from './ProjectCard.js'
    import { CardContextMenu, type ContextMenuAnchor } from './CardContextMenu.js'
    import { RegisterModal } from './RegisterModal.js'
    import { RegisterButtonCard } from './RegisterButtonCard.js'
    import { RenameDialog, EditTagsDialog } from './RenameTagsForms.js'
    import { SchemaDriftState } from './SchemaDriftState.js'
    import { DaemonUnreachableState } from './DaemonUnreachableState.js'
    import type { RegistryListItem } from '@agenticapps/dashboard-shared'

    export function MultiProjectHome(): React.JSX.Element {
      const list = useRegistryList()
      const [selectedChips, setSelectedChips] = useState<Set<string>>(new Set(['all']))
      const [searchText, setSearchText] = useState('')
      const [sortKey, setSortKey] = useState<SortKey>('recommended')

      // Context menu state
      const [menu, setMenu] = useState<{ anchor: ContextMenuAnchor; item: RegistryListItem; initialMode?: 'menu' | 'unregister-confirm' } | null>(null)
      const previouslyFocused = useRef<HTMLElement | null>(null)
      function openMenu(anchor: ContextMenuAnchor, item: RegistryListItem) {
        previouslyFocused.current = document.activeElement as HTMLElement | null
        setMenu({ anchor, item })
      }
      function closeMenu() {
        setMenu(null)
        // restore focus after the menu is unmounted
        setTimeout(() => previouslyFocused.current?.focus(), 0)
      }

      // Modal/dialog states
      const [registerOpen, setRegisterOpen] = useState(false)
      const [renameItem, setRenameItem] = useState<RegistryListItem | null>(null)
      const [tagsItem, setTagsItem] = useState<RegistryListItem | null>(null)
      const [newCardId, setNewCardId] = useState<string | null>(null)

      // Focus on newly-added card after register-confirm closes the modal (D-29)
      useEffect(() => {
        if (!newCardId) return
        const t = setTimeout(() => {
          const card = document.querySelector<HTMLElement>(`[data-card-id="${newCardId}"]`)
          card?.scrollIntoView({ block: 'nearest' })
          card?.focus()
          setNewCardId(null)
        }, 0)
        return () => clearTimeout(t)
      }, [newCardId])

      // Schema drift / 5xx for registry list itself
      if (list.isError) {
        if (list.error?.message?.startsWith('schema_drift:')) {
          return <HomeLayout><SchemaDriftState /></HomeLayout>
        }
        return <HomeLayout><DaemonUnreachableState /></HomeLayout>
      }

      const items = list.data ?? []
      const filtered = filterAndSort(items, { selectedChips, searchText, sortKey })
      const existingTags = Array.from(new Set(items.flatMap(i => i.tags))).sort()

      return (
        <HomeLayout>
          <div className="px-6 py-8 md:px-8">
            <HomeToolbar
              items={items}
              selectedChips={selectedChips}
              onChipsChange={setSelectedChips}
              searchText={searchText}
              onSearchChange={setSearchText}
              sortKey={sortKey}
              onSortChange={setSortKey}
            />
            {items.length === 0 && !list.isLoading ? (
              <div className="mt-6 rounded-md border border-[--border] bg-[--surface] p-6 text-center">
                <h2 className="text-xl font-semibold text-[--text]">No projects registered yet.</h2>
                <p className="mt-2 text-base text-[--text-muted]">
                  Run <code className="font-mono text-sm">agentic-dashboard register &lt;path&gt;</code> to add one, or use the button above.
                </p>
              </div>
            ) : null}
            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 mt-6">
              {filtered.map(item => (
                <div key={item.id} data-card-id={item.id} tabIndex={-1}>
                  <ProjectCard item={item} onContextMenu={openMenu} />
                </div>
              ))}
              <RegisterButtonCard onClick={() => setRegisterOpen(true)} />
            </div>
          </div>
          {menu && (
            <CardContextMenu
              anchor={menu.anchor}
              item={menu.item}
              initialMode={menu.initialMode}
              onAction={(action) => {
                if (action === 'rename') setRenameItem(menu.item)
                else if (action === 'tags') setTagsItem(menu.item)
              }}
              onClose={closeMenu}
            />
          )}
          <RegisterModal
            isOpen={registerOpen}
            onClose={() => setRegisterOpen(false)}
            onConfirmed={(id) => setNewCardId(id)}
          />
          <RenameDialog isOpen={!!renameItem} item={renameItem} onClose={() => setRenameItem(null)} />
          <EditTagsDialog isOpen={!!tagsItem} item={tagsItem} existingTags={existingTags} onClose={() => setTagsItem(null)} />
        </HomeLayout>
      )
    }
    ```
