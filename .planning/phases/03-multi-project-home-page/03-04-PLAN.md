       ```typescript
       registryRoute.post(
         '/register-prepare',
         zValidator('json', RegisterPrepareRequestSchema, (result, c) => {
           if (!result.success) return validationError(c)
         }),
         async (c) => {
           const requestId = (c.get('requestId') as string | undefined) ?? 'unknown'

           // D-14 rate limit (per token, sliding 10s window, cap 10)
           const token = tokenFromAuthHeader(c)
           const tokHash = token ? tokenHashOf(token) : 'no-token'
           const rl = rlConsume(tokHash)
           if (!rl.allowed) {
             return c.json(
               { ok: false, error: 'rate_limited', requestId },
               429,
               { 'Retry-After': String(rl.retryAfter) },
             )
           }

           const body = c.req.valid('json')
           const canonicalRoot = canonicaliseRoot(body.path)
           const registryFile = c.get('registryFile') as string | undefined

           // D-17 already registered?
           const reg = readRegistry(registryFile)
           const existing = reg.projects.find(p => p.root === canonicalRoot)
           if (existing) {
             return outbound(
               c,
               RegisterPrepareResponseSchema.parse.bind(RegisterPrepareResponseSchema),
               { canonicalRoot, alreadyRegistered: true as const, existingEntry: existing },
             )
           }

           // D-11 blocked?
           try {
             assertRegistrationAllowed(canonicalRoot)
           } catch (err) {
             if (err instanceof RegistrationPathBlocked) {
               logBlocked(err.reason, canonicalRoot, tokHash, requestId)  // D-15
               return outbound(
                 c,
                 RegisterPrepareResponseSchema.parse.bind(RegisterPrepareResponseSchema),
                 { canonicalRoot, blocked: true as const, blockedReason: err.reason },
               )
             }
             throw err
           }

           // D-09 issue nonce
           const suggestedName = basename(canonicalRoot)
           const suggestedSlug = slugify(suggestedName)
           const detectedMarkers = detectMarkers(canonicalRoot)
           const { nonce, expiresAt } = issueNonce({ canonicalRoot, suggestedName, suggestedSlug, detectedMarkers })

           return outbound(
             c,
             RegisterPrepareResponseSchema.parse.bind(RegisterPrepareResponseSchema),
             {
               canonicalRoot,
               suggestedName,
               suggestedSlug,
               alreadyRegistered: false as const,
               blocked: false as const,
               detectedMarkers,
               nonce,
               expiresAt,
             },
           )
         },
       )
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| SPA → daemon (HTTP, prepare) | Authenticated bearer holder POSTs `{ path }`. Daemon canonicalises, vets via assertRegistrationAllowed, issues nonce. |
| SPA → daemon (HTTP, confirm) | Authenticated bearer holder POSTs `{ nonce, name?, client?, tags? }`. Daemon consumes nonce, re-vets, registers. |
| Browser tab → token store | Bearer token in localStorage. Compromised token is the dominant threat — D-13 explicitly accepts cross-tab nonce sharing because per-tab nonces don't help once the token is stolen. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-04-01 | T (Tampering) | Confused-deputy register attack | mitigate | D-09 prepare/confirm with nonce. SPA cannot send a path the user didn't preview; the nonce in confirm body is the only thing that authorizes the underlying canonicalRoot. Test "confirm with forged nonce → 410" verifies. |
| T-03-04-02 | E (Elevation of Privilege) | Token replay via stale nonce | mitigate | D-10 single-use (consumeNonce deletes on first use) + D-18 410 on repeat = canonical replay defense. Test "second-confirm-same-nonce → 410" verifies. |
| T-03-04-03 | E (Elevation of Privilege) | Nonce TTL bypass | mitigate | Nonce TTL enforced server-side via `entry.expiresAt < now()` check inside consumeNonce. SPA cannot extend. Test "expired-nonce → 410" verifies. |
| T-03-04-04 | D (Denial of Service) | /register-prepare hammering | mitigate | D-14 rate limiter: 10 calls/10s/token; 11th → 429 + Retry-After: 1. Test "11th call → 429" verifies. |
| T-03-04-05 | D (Denial of Service) | Rate-limit bypass via token rotation | accept | After rotation, the previous tokenHash entries in the rateLimiter Map become orphaned. The new tokenHash starts fresh. Acceptable: rotation is an explicit user action and adds no faster-than-1/s amplification. The 60s sweep cleans up orphans within a minute. |
| T-03-04-06 | T (Tampering) | Path traversal via /register-prepare | mitigate | Phase 1 D-23 realpath defense via canonicaliseRoot (`fs.realpathSync`); assertRegistrationAllowed blocklist on system roots and HOME secret dirs (already shipped in registry.ts). Defense-in-depth on confirm. |
| T-03-04-07 | I (Information Disclosure) | blockedReason leak | accept | blockedReason is intentionally user-visible (D-27) — it tells the user WHY the path was rejected. Reason text never includes secrets; comes from a fixed enumeration in registry.ts. Test "prepare on ~/.ssh → blocked: true + reason verbatim" verifies the verbatim copy. |
| T-03-04-08 | I (Information Disclosure) | Cross-tab nonce confusion | accept | D-13 explicit: nonce bound to bearer token, not per-tab. A malicious in-page script with the token already wins; per-tab nonces add no defense. Documented. |
| T-03-04-09 | R (Repudiation) | Audit log spam from BLOCKED on /register-prepare | accept | logBlocked is bounded by the rateLimiter (max 10/10s/token); no log injection (Wave 0 task 2 sanitizes \\n in reason and root). |
| T-03-04-10 | T (Tampering) | Race between prepare and confirm | mitigate | Confirm re-runs assertRegistrationAllowed (defense-in-depth). If between prepare and confirm something changes (e.g. user manually adds the path to a blocklist, or symlink moves), the second check catches it and returns 422 path_blocked. Test "race window" can be added if a fixture allows simulating mid-flight blocklist mutation; otherwise documented. |
| T-03-04-11 | I (Information Disclosure) | Token leak via tokenHashOf | accept | sha256 first-8-chars is one-way; not a security boundary, it's a log correlation key. Phase 1 D-15 already established the pattern. |
