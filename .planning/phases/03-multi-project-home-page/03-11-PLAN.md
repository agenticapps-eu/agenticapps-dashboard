    ```typescript
    import { describe, it, expect, beforeAll, afterAll } from 'vitest'
    import { spawn, type ChildProcess } from 'node:child_process'
    import { mkdtempSync, readFileSync, mkdirSync, writeFileSync } from 'node:fs'
    import { tmpdir } from 'node:os'
    import { join } from 'node:path'
    import { setTimeout as delay } from 'node:timers/promises'

    let daemon: ChildProcess
    let agentUrl: string
    let token: string
    let homeDir: string
    let projectDir: string

    async function waitForLine(child: ChildProcess, pattern: RegExp, timeoutMs = 10_000): Promise<string> {
      return new Promise((resolve, reject) => {
        const t = setTimeout(() => reject(new Error('timeout waiting for ' + pattern)), timeoutMs)
        const onData = (buf: Buffer) => {
          const line = buf.toString()
          if (pattern.test(line)) {
            clearTimeout(t)
            child.stdout?.off('data', onData)
            resolve(line)
          }
        }
        child.stdout?.on('data', onData)
      })
    }

    beforeAll(async () => {
      homeDir = mkdtempSync(join(tmpdir(), 'agentic-home-'))
      projectDir = mkdtempSync(join(tmpdir(), 'agentic-proj-'))

      daemon = spawn('node', ['packages/agent/dist/cli.js', 'start', '--port', '0', '--bind', '127.0.0.1'], {
        env: { ...process.env, HOME: homeDir, AGENTIC_DASHBOARD_HOME: homeDir, NODE_ENV: 'test' },
        stdio: ['ignore', 'pipe', 'pipe'],
      })

      const banner = await waitForLine(daemon, /Listening on (https?:\/\/\S+)/)
      const m = banner.match(/Listening on (https?:\/\/\S+)/)
      if (!m) throw new Error('failed to parse listening line')
      agentUrl = m[1]!.replace(/\/$/, '')

      // Read fresh token from homeDir/.agenticapps/dashboard/auth.json
      const auth = JSON.parse(readFileSync(join(homeDir, '.agenticapps', 'dashboard', 'auth.json'), 'utf-8'))
      token = auth.token
    }, 30_000)

    afterAll(async () => {
      daemon?.kill('SIGTERM')
      await delay(200)
    })

    describe('Phase 3: register-prepare → register-confirm optimistic-add', () => {
      it('prepares + confirms a registration end-to-end with measurable optimistic-add timing', async () => {
        // Step 1: prepare
        const prepStart = Date.now()
        const prepRes = await fetch(`${agentUrl}/api/registry/register-prepare`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
          body: JSON.stringify({ path: projectDir }),
        })
        expect(prepRes.status).toBe(200)
        const prep = await prepRes.json()
        expect(prep).toHaveProperty('nonce')
        expect(typeof prep.nonce).toBe('string')
        expect(prep.nonce).toMatch(/^[0-9a-f]{32}$/)
        expect(prep.alreadyRegistered).toBe(false)
        expect(prep.blocked).toBe(false)

        // Step 2: confirm
        const confirmRes = await fetch(`${agentUrl}/api/registry/register-confirm`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
          body: JSON.stringify({ nonce: prep.nonce, name: 'phase3-fixture', tags: [] }),
        })
        const totalMs = Date.now() - prepStart
        expect(confirmRes.status).toBe(201)
        const entry = await confirmRes.json()
        expect(entry).toHaveProperty('id')
        expect(entry).toHaveProperty('name', 'phase3-fixture')
        expect(entry).toHaveProperty('alreadyRegistered', false)

        // Acceptance criterion 4: total prepare→confirm < 5s. We measure the daemon round-trip.
        // (The optimistic-add UI tick is already covered in MultiProjectHome.test.tsx.)
        expect(totalMs).toBeLessThan(5000)
      }, 15_000)

      it('rejects confirm with second use of the same nonce (single-use semantics)', async () => {
        const prepRes = await fetch(`${agentUrl}/api/registry/register-prepare`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
          body: JSON.stringify({ path: mkdtempSync(join(tmpdir(), 'agentic-once-')) }),
        })
        const prep = await prepRes.json()
        const first = await fetch(`${agentUrl}/api/registry/register-confirm`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
          body: JSON.stringify({ nonce: prep.nonce }),
        })
        expect(first.status).toBe(201)
        const second = await fetch(`${agentUrl}/api/registry/register-confirm`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
          body: JSON.stringify({ nonce: prep.nonce }),
        })
        expect(second.status).toBe(410)
      }, 15_000)
    })
    ```
