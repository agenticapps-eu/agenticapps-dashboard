    import { useNavigate } from '@tanstack/react-router'
    import { useQueryClient } from '@tanstack/react-query'
    import { useRegistryList } from './registry.js'
    import { useTheme, type ThemeChoice } from './theme.js'

    export interface PaletteAction {
      id: string
      label: string
      type: 'register' | 'refresh' | 'toggle-theme' | 'jump'
      run: () => void
    }

    function nextThemeChoice(current: ThemeChoice): ThemeChoice {
      return current === 'dark' ? 'light' : current === 'light' ? 'system' : 'dark'
    }

    export function useCommandPaletteActions(close: () => void): PaletteAction[] {
      const list = useRegistryList()
      const navigate = useNavigate()
      const qc = useQueryClient()
      const { choice, setChoice } = useTheme()

      const actions: PaletteAction[] = [
        {
          id: 'register',
          label: 'Register project',
          type: 'register',
          run: () => {
            window.dispatchEvent(new CustomEvent('palette:open-register'))
            close()
          },
        },
      ]

      for (const item of list.data ?? []) {
        actions.push({
          id: `jump:${item.id}`,
          label: `Jump to ${item.name}`,
          type: 'jump',
          run: () => {
            navigate({ to: '/projects/$projectId', params: { projectId: item.id } })
            close()
          },
        })
      }

      actions.push({
        id: 'refresh',
        label: 'Refresh data',
        type: 'refresh',
        run: () => {
          qc.invalidateQueries({ queryKey: ['registry'] })
          qc.invalidateQueries({ queryKey: ['overview'] })  // matches all ['overview', *]
          close()
        },
      })

      actions.push({
        id: 'toggle-theme',
        label: 'Toggle theme',
        type: 'toggle-theme',
        run: () => {
          setChoice(nextThemeChoice(choice))
          close()
        },
      })

      return actions
    }

    export function filterActions(actions: PaletteAction[], query: string): PaletteAction[] {
      if (!query) return actions
      const needle = query.toLowerCase()
      return actions.filter(a => a.label.toLowerCase().includes(needle))
    }
    ```

    import { useEffect, useMemo, useRef, useState } from 'react'
    import { Search } from 'lucide-react'
    import { useCommandPaletteActions, filterActions } from '../lib/commandPaletteActions.js'

    export function CommandPalette(): React.JSX.Element {
      const [isOpen, setIsOpen] = useState(false)
      const [query, setQuery] = useState('')
      const [focusedIndex, setFocusedIndex] = useState(0)
      const dialogRef = useRef<HTMLDialogElement>(null)
      const inputRef = useRef<HTMLInputElement>(null)
      const previouslyFocused = useRef<HTMLElement | null>(null)

      const actions = useCommandPaletteActions(() => closeAndRestore())
      const filtered = useMemo(() => filterActions(actions, query), [actions, query])

      function open() {
        previouslyFocused.current = document.activeElement as HTMLElement | null
        setIsOpen(true)
        setQuery('')
        setFocusedIndex(0)
      }
      function closeAndRestore() {
        setIsOpen(false)
        setTimeout(() => previouslyFocused.current?.focus(), 0)
      }

      useEffect(() => {
        function onKey(e: KeyboardEvent) {
          if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
            e.preventDefault()
            open()
          }
        }
        window.addEventListener('keydown', onKey)
        return () => window.removeEventListener('keydown', onKey)
      }, [])

      useEffect(() => {
        if (isOpen) {
          dialogRef.current?.showModal()
          // focus input on next tick
          setTimeout(() => inputRef.current?.focus(), 0)
        } else {
          dialogRef.current?.close()
        }
      }, [isOpen])

      // clamp focusedIndex when filter changes
      useEffect(() => {
        if (focusedIndex >= filtered.length) setFocusedIndex(Math.max(0, filtered.length - 1))
      }, [filtered.length, focusedIndex])

      function handleKeyDown(e: React.KeyboardEvent) {
        if (e.key === 'ArrowDown') {
          e.preventDefault()
          setFocusedIndex(i => filtered.length === 0 ? 0 : (i + 1) % filtered.length)
        } else if (e.key === 'ArrowUp') {
          e.preventDefault()
          setFocusedIndex(i => filtered.length === 0 ? 0 : (i - 1 + filtered.length) % filtered.length)
        } else if (e.key === 'Enter') {
          e.preventDefault()
          if (filtered[focusedIndex]) filtered[focusedIndex].run()
        } else if (e.key === 'Tab') {
          e.preventDefault()
          closeAndRestore()
        }
      }

      const activeId = filtered.length > 0 && focusedIndex < filtered.length
        ? `palette-option-${focusedIndex}` : undefined

      return (
        <dialog
          ref={dialogRef}
          onCancel={(e) => { e.preventDefault(); closeAndRestore() }}
          onClose={() => setIsOpen(false)}
          className="bg-[--surface] border border-[--border-strong] rounded-lg overflow-hidden p-0 max-w-lg w-full mt-[20vh] mx-auto backdrop:bg-black/60"
        >
          <div className="flex flex-col" onKeyDown={handleKeyDown}>
            <div className="relative flex items-center border-b border-[--border]">
              <Search size={16} aria-hidden="true" className="absolute left-4 text-[--text-muted]" />
              <input
                ref={inputRef}
                type="search"
                aria-label="Command palette search"
                aria-owns="palette-listbox"
                aria-activedescendant={activeId}
                placeholder="Search or jump to…"
                value={query}
                onChange={(e) => setQuery(e.target.value)}
                className="w-full bg-transparent border-0 pl-12 pr-16 py-3 text-base text-[--text] placeholder:text-[--text-subtle] focus:outline-none"
              />
              <span className="absolute right-4 bg-[--surface-elevated] border border-[--border] rounded px-2 leading-none flex items-center font-mono text-xs text-[--text-muted]" aria-hidden="true">⌘K</span>
            </div>
            <ul id="palette-listbox" role="listbox" aria-label="Actions" className="max-h-72 overflow-y-auto">
              {filtered.length === 0 ? (
                <li className="px-4 py-3 text-sm text-[--text-muted]">No actions found. Try a shorter search.</li>
              ) : (
                filtered.map((a, i) => (
                  <li
                    key={a.id}
                    id={`palette-option-${i}`}
                    role="option"
                    aria-selected={i === focusedIndex}
                    onMouseEnter={() => setFocusedIndex(i)}
                    onClick={() => a.run()}
                    className={`flex items-center gap-3 px-4 py-3 text-sm text-[--text] cursor-pointer ${i === focusedIndex ? 'bg-[--surface-elevated]' : ''}`}
                  >
                    {a.label}
                  </li>
                ))
              )}
            </ul>
          </div>
        </dialog>
      )
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Window key handler → palette open | Synthetic events from extensions can fire keydown; the palette open is a UI-only side effect (no data exfil). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-10-01 | T (Tampering) | Synthetic Cmd+K from a malicious script | accept | Opening the palette is harmless; the user must explicitly Enter on a row to fire an action. Same posture as any keyboard shortcut. |
| T-03-10-02 | I (Information Disclosure) | Project names visible in palette | accept | Bearer-gated app surface; user already sees them on the home page. |
| T-03-10-03 | T (Tampering) | XSS via project name in Jump-to row | mitigate | Action labels are constructed via template-literal `\`Jump to ${name}\`` and rendered via React `{a.label}` text rendering — no raw HTML. |
| T-03-10-04 | E (Elevation of Privilege) | Action emit-event design | accept | `palette:open-register` CustomEvent on window has no payload; only opens a UI surface that already exists. No new auth boundary. |
| T-03-10-05 | D (Denial of Service) | global keydown listener | mitigate | Single listener; ~5 lines of work per keystroke; no DoS surface. |
</threat_model>
