```typescript
import { existsSync, readFileSync, readdirSync, realpathSync, statSync } from 'node:fs'
import { join, sep } from 'node:path'

export interface SkillFrontmatter { /* per <interfaces> */ }
export interface SkillEntry extends SkillFrontmatter { /* per <interfaces> */ }

const FRONTMATTER_RE = /^---\n([\s\S]*?)\n---/m

export function parseFrontmatter(skillMdPath: string): SkillFrontmatter | null {
  let raw: string
  try { raw = readFileSync(skillMdPath, 'utf8') } catch { return null }
  const m = raw.match(FRONTMATTER_RE)
  if (!m) return null
  const body = m[1]!
  const out: Record<string, unknown> = {}
  const lines = body.split('\n')
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i]!
    const colonIdx = line.indexOf(':')
    if (colonIdx === -1) continue
    const key = line.slice(0, colonIdx).trim()
    let val = line.slice(colonIdx + 1).trim()
    if (!key) continue
    if (val === '|') {
      // YAML literal block — read subsequent indented lines
      const blockLines: string[] = []
      let j = i + 1
      while (j < lines.length) {
        const next = lines[j]!
        if (next.length === 0) { blockLines.push(''); j++; continue }
        if (!/^\s/.test(next)) break  // non-indented = end of block
        blockLines.push(next.replace(/^\s+/, ''))
        j++
      }
      out[key] = blockLines.join('\n').trimEnd()
      i = j - 1
    } else {
      out[key] = val
    }
  }
  if (typeof out.name !== 'string') {
    const parts = skillMdPath.split(sep)
    // For canonical layout (.../<skillDir>/SKILL.md), name = parts[-2]
    // For bundle layout (.../<skillDir>/skill/SKILL.md), name = parts[-3]
    out.name = parts[parts.length - 2] === 'skill' ? (parts[parts.length - 3] ?? 'unknown') : (parts[parts.length - 2] ?? 'unknown')
  }
  return out as SkillFrontmatter
}

export async function readGlobalSkills(root: string): Promise<{ scope: 'global'; skills: SkillEntry[] }> {
  return readSkillsAt(root, 'global')
}

export async function readLocalSkills(projectRoot: string): Promise<{ scope: 'local'; skills: SkillEntry[] }> {
  return readSkillsAt(join(projectRoot, '.claude', 'skills'), 'local')
}

async function readSkillsAt(root: string, scope: 'global' | 'local'): Promise<any> {
  if (!existsSync(root)) return { scope, skills: [] }
  let realRoot: string
  try { realRoot = realpathSync(root) } catch { return { scope, skills: [] } }

  let entries: string[]
  try { entries = readdirSync(realRoot) } catch { return { scope, skills: [] } }
  entries.sort()

  const skills: SkillEntry[] = []
  for (const dir of entries) {
    const skillRoot = join(realRoot, dir)
    let real: string
    try {
      real = realpathSync(skillRoot)
    } catch { continue }
    // Symlink-escape defense: realpath must be inside realRoot
    if (real !== skillRoot && !real.startsWith(realRoot + sep)) continue
    let stat
    try { stat = statSync(real) } catch { continue }
    if (!stat.isDirectory()) continue

    // Probe canonical, then bundle — REUSES Phase 4 phaseDetail.ts:129-137 pattern
    const canonical = join(real, 'SKILL.md')
    const bundle = join(real, 'skill', 'SKILL.md')
    let mdPath: string | null = null
    if (existsSync(canonical)) mdPath = canonical
    else if (existsSync(bundle)) mdPath = bundle
    if (!mdPath) continue

    const fm = parseFrontmatter(mdPath)
    if (!fm) continue

    skills.push({ ...fm, dir, scope } as SkillEntry)
  }
  return { scope, skills }
}
```

import type { AgentLinterReport } from '@agenticapps/dashboard-shared'

const TIMEOUT_MS = 30_000

export type AgentLinterResult =
  | { kind: 'ok'; report: AgentLinterReport }
  | { kind: 'not-installed' }
  | { kind: 'timeout' }
  | { kind: 'error'; exitCode: number; stderr: string }
  | { kind: 'unparseable'; exitCode: number; rawStdout: string }

export async function runAgentLinter(projectRoot: string): Promise<AgentLinterResult> {
  let proc
  try {
    proc = await execa(
      'npx',
      ['--yes', 'agentlinter', '--local', '--json', projectRoot],
      {
        timeout: TIMEOUT_MS,
        reject: false,
        stdio: ['ignore', 'pipe', 'pipe'],
      },
    )
  } catch (e: unknown) {
    const err = e as { timedOut?: boolean }
    if (err?.timedOut) return { kind: 'timeout' }
    return { kind: 'not-installed' }
  }
  if ((proc as any).timedOut) return { kind: 'timeout' }
  const exitCode = proc.exitCode ?? -1
  const stderr = proc.stderr ?? ''
  const stdout = proc.stdout ?? ''
  if (exitCode !== 0) {
    if (/E404|not found/i.test(stderr)) return { kind: 'not-installed' }
    // Some agentlinter errors still emit JSON envelope on stdout
    try {
      const parsed = JSON.parse(stdout)
      return { kind: 'ok', report: parsed }
    } catch { /* fall through */ }
    return { kind: 'error', exitCode, stderr }
  }
  try {
    return { kind: 'ok', report: JSON.parse(stdout) }
  } catch {
    return { kind: 'unparseable', exitCode, rawStdout: stdout }
  }
}

const CACHE_TTL_MS = 3_600_000  // 1h hard ceiling per D-5-14

  cachedAt: string  // ISO
  maxMtime: number  // ms epoch

export async function computeMaxMtime(projectRoot: string, globalRoot?: string): Promise<number> {
  const roots = [
    join(projectRoot, '.claude', 'skills'),
    globalRoot ?? join(homedir(), '.claude', 'skills'),
  ]

  for (const root of roots) {

    let dirs: string[]
    try { dirs = await readdir(root) } catch { continue }
    for (const dir of dirs) {
      const canonical = join(root, dir, 'SKILL.md')
      const bundle = join(root, dir, 'skill', 'SKILL.md')
      for (const candidate of [canonical, bundle]) {
        if (!existsSync(candidate)) continue
        try {
          const s = statSync(candidate)
          if (s.mtimeMs > maxM) maxM = s.mtimeMs
        } catch { /* skip */ }
      }
    }
  }
  return maxM
}
