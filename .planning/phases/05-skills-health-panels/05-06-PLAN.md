```javascript
#!/usr/bin/env node
// Phase 5 D-5-10 closure-gate scripted half.
// Round-trips meta-observer producer (session-end.mjs) → consumer (Phase 4 parseCommitmentBlock + readSkillObservations).

import assert from 'node:assert/strict'
import { spawn } from 'node:child_process'
import { mkdirSync, mkdtempSync, readdirSync, readFileSync, rmSync, writeFileSync } from 'node:fs'
import { tmpdir } from 'node:os'
import { dirname, join, resolve } from 'node:path'
import { fileURLToPath } from 'node:url'

const here = dirname(fileURLToPath(import.meta.url))
const repoRoot = resolve(here, '..', '..', '..')
const hookScript = join(repoRoot, 'packages', 'meta-observer', 'hooks', 'session-end.mjs')
const fixtureTranscript = join(here, '__fixtures__', 'sample-transcript.jsonl')

async function main() {
  // 1. tmp project root
  const tmpRoot = mkdtempSync(join(tmpdir(), 'meta-observer-e2e-'))
  console.log(`[e2e] tmp root: ${tmpRoot}`)
  try {
    mkdirSync(join(tmpRoot, '.planning', 'skill-observations'), { recursive: true })
    mkdirSync(join(tmpRoot, '.claude', 'skills', 'meta-observer'), { recursive: true })
    // Copy SKILL.md so the consumer's findSkillPath() finds the skill (DISC-04 install hint won't fire)
    const skillMd = readFileSync(join(repoRoot, 'packages', 'meta-observer', 'SKILL.md'), 'utf8')
    writeFileSync(join(tmpRoot, '.claude', 'skills', 'meta-observer', 'SKILL.md'), skillMd)

    // 2. SessionEnd payload
    const payload = {
      session_id: 'e2e-test-001',
      transcript_path: fixtureTranscript,
      cwd: tmpRoot,
      hook_event_name: 'SessionEnd',
    }

    // 3. Spawn hook
    const proc = spawn('node', [hookScript], { stdio: ['pipe', 'pipe', 'pipe'] })
    proc.stdin.write(JSON.stringify(payload))
    proc.stdin.end()

    let stderr = ''
    proc.stderr.on('data', (d) => { stderr += d.toString() })

    const exitCode = await new Promise((resolve) => proc.on('close', resolve))
    assert.equal(exitCode, 0, `hook exited non-zero. stderr: ${stderr}`)

    // 4. Assert files
    const obsDir = join(tmpRoot, '.planning', 'skill-observations')
    const entries = readdirSync(obsDir).filter((f) => f.includes('e2e-test-001'))
    console.log(`[e2e] observations: ${entries.join(', ')}`)
    const mdFiles = entries.filter((f) => f.endsWith('.md'))
    const jsonlFiles = entries.filter((f) => f.endsWith('.jsonl'))
    assert.ok(mdFiles.length >= 1 || jsonlFiles.length >= 1, 'expected at least one .md or .jsonl output (fixture may have no commitment + no firings)')

    // 5. Validate .jsonl against HookFiringSchema (dynamic import — workspace package)
    if (jsonlFiles.length > 0) {
      const { HookFiringSchema } = await import('@agenticapps/dashboard-shared')
      for (const jl of jsonlFiles) {
        const content = readFileSync(join(obsDir, jl), 'utf8')
        const lines = content.split('\n').filter((l) => l.trim().length > 0)
        for (const line of lines) {
          const parsed = JSON.parse(line)
          const valid = HookFiringSchema.safeParse(parsed)
          assert.ok(valid.success, `JSONL line failed schema validation: ${line.slice(0, 80)}`)
        }
        console.log(`[e2e] ${jl} → ${lines.length} valid firings`)
      }
    }

    // 6. Round-trip via Phase 4 consumer
    const { parseCommitmentBlock, readSkillObservations } = await import('@agenticapps/dashboard-agent/dist/lib/phaseDetail.js').catch(async () => {
      // Fallback: if dist not built, try src
      return import(join(repoRoot, 'packages', 'agent', 'src', 'lib', 'phaseDetail.ts'))
    })
    const commitment = await parseCommitmentBlock(obsDir)
    const observations = await readSkillObservations(tmpRoot, 20)
    console.log(`[e2e] commitment markdown: ${commitment.markdown ? 'present (' + commitment.markdown.slice(0, 60) + '...)' : 'null'}`)
    console.log(`[e2e] observations: skillInstalled=${observations.skillInstalled}, entries=${observations.entries.length}`)

    // Either commitment OR observations must be non-empty (fixture-driven)
    assert.ok(commitment.markdown !== null || observations.entries.length > 0, 'no extracted commitment AND no extracted firings — fixture insufficient')

    console.log('[e2e] PASS')
  } finally {
    rmSync(tmpRoot, { recursive: true, force: true })
  }
}

main().catch((e) => {
  console.error('[e2e] FAIL:', e)
  process.exit(1)
})
```
