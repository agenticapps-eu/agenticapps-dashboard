For each tool's `signals` array, build the provenance string by joining the evidence strings with ` + ` (or, for cleaner UX, by mapping signal enum → human label — pick one approach in the implementation; the simpler `signals.map(s => s.evidence).join(' + ')` is fine).

```typescript
import React from 'react'

import { useObservability } from '../../lib/projectQueries.js'

import { PanelContainer } from './PanelContainer.js'
import { InlineDrift } from './InlineDrift.js'

export type ObservabilityHealthProps = { projectId: string }

const PANEL_ID = 'observability-health'
const PANEL_TITLE = 'Observability'

export function ObservabilityHealth({ projectId }: ObservabilityHealthProps): React.JSX.Element {
  const query = useObservability(projectId)

  if (query.error?.message?.startsWith('schema_drift:')) {
    return <InlineDrift panelId={PANEL_ID} title={PANEL_TITLE} path={query.error.message.slice('schema_drift:'.length)} onRetry={() => query.refetch()} />
  }
  if (query.isLoading) {
    return <PanelContainer panelId={PANEL_ID} title={PANEL_TITLE}><p className="text-sm text-[--text-muted]">Loading...</p></PanelContainer>
  }
  if (query.error || !query.data) {
    return <PanelContainer panelId={PANEL_ID} title={PANEL_TITLE} unreachable>{null}</PanelContainer>
  }

  const data = query.data
  const tools: Array<[string, typeof data.sentry]> = [['Sentry', data.sentry], ['Spotlight', data.spotlight], ['sentry-cli', data.sentryCli]]
  const allEmpty = tools.every(([_, t]) => !t.detected)

  if (allEmpty) {
    return (
      <PanelContainer panelId={PANEL_ID} title={PANEL_TITLE}>
        <p className="text-base leading-relaxed text-[--text-muted]">No observability tooling detected. (Configure to enable.)</p>
      </PanelContainer>
    )
  }

  return (
    <PanelContainer panelId={PANEL_ID} title={PANEL_TITLE}>
      <div className="grid grid-cols-[8rem_1fr] gap-3">
        {tools.map(([name, tool]) => (
          <React.Fragment key={name}>
            <span className="text-sm font-semibold text-[--text]">{name}</span>
            {tool.detected ? (
              <span>
                <span className="text-sm text-[--text-muted]">detected via </span>
                <span className="text-sm text-[--text]">{tool.signals.map((s) => s.evidence).join(' + ')}</span>
              </span>
            ) : (
              <span className="text-sm text-[--text-muted] italic">not detected</span>
            )}
          </React.Fragment>
        ))}
      </div>
    </PanelContainer>
  )
}
```

import { useSecrets } from '../../lib/projectQueries.js'

import { PanelContainer } from './PanelContainer.js'
import { InlineDrift } from './InlineDrift.js'

export type SecretsHealthProps = { projectId: string }

const PANEL_ID = 'secrets-health'
const PANEL_TITLE = 'Secrets'

export function SecretsHealth({ projectId }: SecretsHealthProps): React.JSX.Element {
  const query = useSecrets(projectId)
  // Loading / drift / unreachable handled as above (omitted — same pattern)
  if (query.error?.message?.startsWith('schema_drift:')) {
    return <InlineDrift panelId={PANEL_ID} title={PANEL_TITLE} path={query.error.message.slice('schema_drift:'.length)} onRetry={() => query.refetch()} />
  }
  if (query.isLoading) {

  }

  return (

        <div className="flex items-center gap-2">

        </div>
      )}

  )
}
