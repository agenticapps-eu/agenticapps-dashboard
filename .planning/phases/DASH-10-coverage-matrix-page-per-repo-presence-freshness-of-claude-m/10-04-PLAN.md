**coverage.ts** — Implement per the Pattern 1 RESEARCH example. Key structure:

```typescript
import { Hono } from 'hono'
import { zValidator } from '@hono/zod-validator'  // if not present, fallback to manual c.req.json() + safeParse
import {
  CoverageResponseSchema,
  CoverageRefreshRequestSchema,
  CoverageRefreshResponseSchema,
  type CoverageRefreshResponse,
} from '@agenticapps/dashboard-shared'

import { scanCoverage } from '../lib/coverageScan.js'
import {
  getCoverageCache,
  setCoverageCache,
  invalidateCoverageCache,
} from '../lib/coverageCache.js'
import { spawnGitNexusAnalyze } from '../lib/coverageSpawn.js'
import { outbound } from '../server/middleware/errors.js'
import type { Env } from '../server/app.js'

export const coverageRoute = new Hono<Env>()

// ── GET /coverage ──────────────────────────────────────────────────────────
coverageRoute.get('/coverage', async (c) => {
  const cached = getCoverageCache()
  if (cached) {
    return outbound(c, CoverageResponseSchema.parse.bind(CoverageResponseSchema), cached)
  }
  const value = await scanCoverage()
  setCoverageCache(value)
  return outbound(c, CoverageResponseSchema.parse.bind(CoverageResponseSchema), value)
})

// ── POST /coverage/refresh ─────────────────────────────────────────────────
coverageRoute.post('/coverage/refresh', async (c) => {
  // Validate body via the shared schema. zValidator handles 400 on failure, OR
  // do this manually:
  let body
  try {
    body = CoverageRefreshRequestSchema.parse(await c.req.json())
  } catch (e) {
    // D-10-09: action='wiki-compile' fails here (not in enum)
    return c.json({ error: 'invalid request body', detail: String(e) }, 400)
  }

  // Single safe action in v1.0 (D-10-09): gitnexus-analyze only
  // (Schema parse already ensures action === 'gitnexus-analyze'; defensive switch retained
  // so adding new safe actions in a future migration is mechanical.)
  let resp: CoverageRefreshResponse
  switch (body.action) {
    case 'gitnexus-analyze': {
      // Look up the discovered repo's absolute path. For v1, derive it from family + repo
      // (the route trusts the SPA's {family,repo} pair; the spawn module's cwd is the absPath).
      // Reuse the scanCoverage discovery → find row → use absPath.
      const matrix = await scanCoverage()
      const row = matrix.rows.find((r) => r.family === body.family && r.repo === body.repo)
      if (!row) {
        return c.json({ error: 'repo not found', family: body.family, repo: body.repo }, 404)
      }
      const result = await spawnGitNexusAnalyze(row.absPath)
      resp = mapSpawnToResponse(result, row)
      break
    }
    // No other actions accepted (D-10-09 — wiki, claude-md, workflow-update are clipboard-only)
  }
  invalidateCoverageCache()
  return outbound(c, CoverageRefreshResponseSchema.parse.bind(CoverageRefreshResponseSchema), resp)
})

function mapSpawnToResponse(
  result: Awaited<ReturnType<typeof spawnGitNexusAnalyze>>,
  row: CoverageRow,
): CoverageRefreshResponse {
  if (result.kind === 'ok') {
    // Updated row is best-effort; the SPA will re-fetch /coverage and the cleared
    // cache means the next GET returns fresh data. We omit updatedRow here to keep
    // the response narrow (SPA refetches via TanStack Query mutation onSuccess).
    return { ok: true, kind: 'ok' }
  }
  if (result.kind === 'not-installed') return { ok: false, kind: 'not-installed' }
  if (result.kind === 'timeout') return { ok: false, kind: 'timeout' }
  return { ok: false, kind: 'error', exitCode: result.exitCode, stderr: result.stderr }
}
```
