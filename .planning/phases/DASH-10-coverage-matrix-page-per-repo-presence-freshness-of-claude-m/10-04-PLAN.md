```typescript
import { Hono } from 'hono'
import { zValidator } from '@hono/zod-validator'  // if not present, fallback to manual c.req.json() + safeParse
import {
  CoverageResponseSchema,
  CoverageRefreshRequestSchema,
  CoverageRefreshResponseSchema,
  type CoverageRefreshResponse,
} from '@agenticapps/dashboard-shared'

import { scanCoverage } from '../lib/coverageScan.js'
import {
  getCoverageCache,
  setCoverageCache,
  invalidateCoverageCache,
} from '../lib/coverageCache.js'
import { spawnGitNexusAnalyze } from '../lib/coverageSpawn.js'
import { outbound } from '../server/middleware/errors.js'
import type { Env } from '../server/app.js'

export const coverageRoute = new Hono<Env>()

// ── GET /coverage ──────────────────────────────────────────────────────────
coverageRoute.get('/coverage', async (c) => {
  const cached = getCoverageCache()
  if (cached) {
    return outbound(c, CoverageResponseSchema.parse.bind(CoverageResponseSchema), cached)
  }
  const value = await scanCoverage()
  setCoverageCache(value)
  return outbound(c, CoverageResponseSchema.parse.bind(CoverageResponseSchema), value)
})

// CODEX MED-14: per-repo refresh lock — concurrent POSTs against the same {family,repo}
// serialise on the same promise. Resolved promises are cleaned up so the map stays bounded.
const refreshLocks = new Map<string, Promise<CoverageRefreshResponse>>()

// ── POST /coverage/refresh ─────────────────────────────────────────────────
coverageRoute.post('/coverage/refresh', async (c) => {
  // Validate body via the shared schema (D-10-09 + CODEX HIGH-5: only 'gitnexus-analyze' is valid)
  let body
  try {
    body = CoverageRefreshRequestSchema.parse(await c.req.json())
  } catch (e) {
    return c.json({ error: 'invalid request body', detail: String(e) }, 400)
  }

  const lockKey = `${body.family}/${body.repo}`
  // CODEX MED-14: if a refresh for this {family,repo} is already in flight, await it.
  const existing = refreshLocks.get(lockKey)
  if (existing) {
    const resp = await existing
    return outbound(c, CoverageRefreshResponseSchema.parse.bind(CoverageRefreshResponseSchema), resp)
  }

  const inflight = (async (): Promise<CoverageRefreshResponse> => {
    // AGREED-3: resolve {family,repo} → absPath via SYNCHRONOUS discoverRepos() — not scanCoverage().
    const repos = discoverRepos()
    const match = repos.find((r) => r.family === body.family && r.name === body.repo)
    if (!match) {
      return { ok: false, kind: 'error', exitCode: -1, stderr: 'repo not found in discovery set' }
    }

    // CODEX HIGH-3 TOCTOU mitigation: re-canonicalise absPath via realpathSync IMMEDIATELY
    // before spawn. Re-assert that the realpath stays under the family root + sep.
    let canonicalAbs: string
    try {
      canonicalAbs = realpathSync(match.absPath)
    } catch {
      return { ok: false, kind: 'error', exitCode: -1, stderr: 'repo no longer accessible' }
    }
    const familyRoot = realpathSync(join(homedir(), 'Sourcecode', body.family))
    if (!(canonicalAbs === familyRoot || canonicalAbs.startsWith(familyRoot + sep))) {
      return { ok: false, kind: 'error', exitCode: -1, stderr: 'repo path escapes family root after canonicalisation (TOCTOU)' }
    }

    // Single safe action: gitnexus-analyze
    const result = await spawnGitNexusAnalyze(canonicalAbs)
    invalidateCoverageCache()  // clear cache so the next scan re-reads

    if (result.kind === 'ok') {
      // CODEX HIGH-5: updatedRow REQUIRED on success. Re-scan internally; lookup the row.
      const { response: fresh } = await scanCoverageInternal()
      const updatedRow = fresh.rows.find((r) => r.family === body.family && r.repo === body.repo)
      if (!updatedRow) {
        // Should never happen — the row was discovered above. Defensive return.
        return { ok: false, kind: 'error', exitCode: -1, stderr: 'updated row missing after scan' }
      }
      return { ok: true, kind: 'ok', updatedRow }
    }
    if (result.kind === 'not-installed') return { ok: false, kind: 'not-installed' }
    if (result.kind === 'timeout') return { ok: false, kind: 'timeout' }
    return { ok: false, kind: 'error', exitCode: result.exitCode, stderr: result.stderr }
  })()
  // CODEX MED-14: lock for the lifetime of the spawn; clean up after settle.
  refreshLocks.set(lockKey, inflight)
  try {
    const resp = await inflight
    return outbound(c, CoverageRefreshResponseSchema.parse.bind(CoverageRefreshResponseSchema), resp)
  } finally {
    refreshLocks.delete(lockKey)
  }
})
```
