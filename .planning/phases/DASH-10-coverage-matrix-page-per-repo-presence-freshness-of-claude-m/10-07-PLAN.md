```typescript
import { test, expect } from '@playwright/test'

test.describe('/coverage user journey', () => {
  test.beforeEach(async ({ page }) => {
    // Pair the SPA (use existing test-helper or localStorage seed)
    await page.goto('/')
    await page.evaluate(() => {
      localStorage.setItem('dashboard:pairing', JSON.stringify({
        agentUrl: 'http://127.0.0.1:5193',
        token: 'test-token',
      }))
    })
  })

  test('cold-load: matrix renders 3 family sections', async ({ page }) => {
    await page.goto('/coverage')
    await expect(page.getByRole('heading', { name: 'Coverage' })).toBeVisible()
    await expect(page.getByText('agenticapps')).toBeVisible()
    await expect(page.getByText('factiv')).toBeVisible()
    await expect(page.getByText('neuroflash')).toBeVisible()
  })

  test('filter chip click — selecting "missing" hides fresh-only rows', async ({ page }) => {
    await page.goto('/coverage')
    await page.getByRole('button', { name: /missing/i }).click()
    // Assert URL has ?status=missing
    await expect(page).toHaveURL(/status=missing/)
  })

  test('search input — typing "agent" filters to repos containing "agent"', async ({ page }) => {
    await page.goto('/coverage')
    await page.getByLabel(/search repos/i).fill('agent')
    await page.waitForTimeout(250)  // 200ms debounce + buffer
    await expect(page).toHaveURL(/q=agent/)
  })

  test('override chip — click expands inline list', async ({ page }) => {
    // Note: zero sentinels exist by default — this test only runs if at least one row has overrideCount > 0.
    await page.goto('/coverage')
    const chip = page.getByRole('button', { name: /override/i }).first()
    if (await chip.count() > 0) {
      await chip.click()
      await expect(chip).toHaveAttribute('aria-expanded', 'true')
    } else {
      test.skip(true, 'No override sentinels on this machine — chip not rendered (Pitfall 5 — by design)')
    }
  })

  test('refresh popover — opens on action button click, Esc dismisses', async ({ page }) => {
    await page.goto('/coverage')
    const firstRefreshBtn = page.getByRole('button', { name: /refresh.*for/i }).first()
    if (await firstRefreshBtn.count() > 0) {
      await firstRefreshBtn.click()
      await page.keyboard.press('Escape')
      // Popover should be dismissed
    }
  })

  test('keyboard navigation — Tab through page-header → toolbar → first family section', async ({ page }) => {
    await page.goto('/coverage')
    await page.keyboard.press('Tab')
    await page.keyboard.press('Tab')
    await page.keyboard.press('Tab')
    // Assert focused element is one of the expected targets
    const focused = await page.evaluate(() => document.activeElement?.tagName)
    expect(['BUTTON', 'INPUT', 'A']).toContain(focused)
  })

  test('GitNexus not-installed banner — shown when daemon reports gitNexusInstalled=false', async ({ page }) => {
    await page.goto('/coverage')
    // On current dev machine, gitnexus is absent → banner SHOULD render
    const banner = page.getByText(/GitNexus is not installed/i)
    await expect(banner).toBeVisible({ timeout: 5000 })
  })
})
```
