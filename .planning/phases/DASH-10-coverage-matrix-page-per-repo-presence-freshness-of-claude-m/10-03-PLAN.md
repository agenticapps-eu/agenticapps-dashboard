```typescript
import { execa } from 'execa'
import { execFile } from 'node:child_process'
import { promisify } from 'node:util'

const SPAWN_TIMEOUT_MS = 5 * 60 * 1000

const execFileP = promisify(execFile)

async function resolveOnPath(cmd: string): Promise<string | null> {
  try {
    // argv-array form — no shell, no string interpolation
    const { stdout } = await execFileP('which', [cmd])  // POSIX; on Windows use 'where' but project targets macOS/Linux
    const trimmed = stdout.trim().split('\n')[0]
    return trimmed || null
  } catch {
    return null
  }
}

export type SpawnResult =
  | { kind: 'ok'; stdout: string }
  | { kind: 'not-installed' }
  | { kind: 'timeout' }
  | { kind: 'error'; exitCode: number; stderr: string }

export async function spawnGitNexusAnalyze(repoAbsPath: string): Promise<SpawnResult> {
  const cmd = await resolveOnPath('gitnexus')
  if (!cmd) return { kind: 'not-installed' }  // D-5-21: never `npx gitnexus`
  try {
    const result = await execa(cmd, ['analyze'], {
      cwd: repoAbsPath,
      timeout: SPAWN_TIMEOUT_MS,
    })
```

```typescript
import { realpathSync } from 'node:fs'
import { resolve as pathResolve, sep, basename } from 'node:path'
import { homedir } from 'node:os'

export class PathViolation extends Error {
  constructor(message: string) { super(message); this.name = 'PathViolation' }
}

export type PathResolver = (
  candidatePath: string,
  opts: { allowedNames?: string[]; extension?: string; roots: string[] },
) => string

export interface CoverageResolverOptions {
  sourcecodeRoot?: string
  gitnexusHome?: string
  migrationsDir?: string
}

export function makeCoverageResolver(opts: CoverageResolverOptions = {}): PathResolver {
  // Pre-realpath the well-known roots so scanners can pass any of them.
  // We don't restrict opts.roots here — the scanner picks the relevant subset per call.
  return (candidatePath, callOpts) => {
    if (callOpts.allowedNames && callOpts.extension) {
      throw new PathViolation('opts.allowedNames and opts.extension are mutually exclusive')
    }
    if (!callOpts.allowedNames && !callOpts.extension) {
      throw new PathViolation('one of opts.allowedNames or opts.extension is required')
    }
    let real: string
    try {
      real = realpathSync(candidatePath)
    } catch {
      throw new PathViolation('not accessible')
    }
    const realRoots = callOpts.roots.map((r) => {
      try { return realpathSync(r) } catch { return pathResolve(r) }
    })
    const inRoot = realRoots.some((root) => real === root || real.startsWith(root + sep))
    if (!inRoot) throw new PathViolation('outside allowed roots')
    const name = basename(real)
    if (callOpts.allowedNames && !callOpts.allowedNames.includes(name)) {
      throw new PathViolation(`name not in allow-list: ${name}`)
    }
    if (callOpts.extension && !name.endsWith(callOpts.extension)) {
      throw new PathViolation(`extension not allowed: ${name}`)
    }
    return real
  }
}
```
