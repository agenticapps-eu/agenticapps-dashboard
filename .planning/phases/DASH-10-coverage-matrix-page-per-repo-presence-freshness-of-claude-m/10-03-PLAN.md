```typescript
import type { CoverageResponse } from '@agenticapps/dashboard-shared'

export interface CoverageCacheEntry {
  value: CoverageResponse
  expiresAt: number
}

export const TTL_MS = 30_000

let cache: CoverageCacheEntry | null = null

export function getCoverageCache(now: number = Date.now()): CoverageResponse | null {
  if (!cache) return null
  if (now >= cache.expiresAt) return null
  return cache.value
}

  cache = { value, expiresAt: now + TTL_MS }
}

  cache = null
}

}
```

```typescript
import { execa } from 'execa'
import { execFile } from 'node:child_process'
import { promisify } from 'node:util'

const SPAWN_TIMEOUT_MS = 5 * 60 * 1000

const execFileP = promisify(execFile)

async function resolveOnPath(cmd: string): Promise<string | null> {
  try {
    // argv-array form — no shell, no string interpolation
    const { stdout } = await execFileP('which', [cmd])  // POSIX; on Windows use 'where' but project targets macOS/Linux
    const trimmed = stdout.trim().split('\n')[0]
    return trimmed || null
  } catch {
    return null
  }
}

export type SpawnResult =
  | { kind: 'ok'; stdout: string }
  | { kind: 'not-installed' }
  | { kind: 'timeout' }
  | { kind: 'error'; exitCode: number; stderr: string }

export async function spawnGitNexusAnalyze(repoAbsPath: string): Promise<SpawnResult> {
  const cmd = await resolveOnPath('gitnexus')
  if (!cmd) return { kind: 'not-installed' }  // D-5-21: never `npx gitnexus`
  try {
    const result = await execa(cmd, ['analyze'], {
      cwd: repoAbsPath,
      timeout: SPAWN_TIMEOUT_MS,
    })
    return { kind: 'ok', stdout: result.stdout }
  } catch (e: any) {
    if (e.timedOut) return { kind: 'timeout' }
    return { kind: 'error', exitCode: e.exitCode ?? -1, stderr: e.stderr ?? '' }
  }
}

export function buildWikiCompileClipboardString(
  family: 'agenticapps' | 'factiv' | 'neuroflash'
): string {
  // D-10-09: wiki refresh is CLIPBOARD-ONLY in v1.0; daemon NEVER spawns wiki-compile.
  return `cd ~/Sourcecode/${family} && claude /wiki-compile`
}

  return '/help/operations/install#claude-md-bootstrap'
}

export function buildWorkflowUpdateClipboardString(): string {

}
```

```typescript
import { z } from 'zod'
import { discoverRepos, type DiscoveredRepo } from './repoDiscovery.js'
import { scanClaudeMd } from './scanners/claudeMdScanner.js'
import { scanGitNexusGlobal, rateGitNexusRepo } from './scanners/gitNexusScanner.js'
import { scanWikiForFamily } from './scanners/wikiScanner.js'
import {
  readWorkflowHeadVersion,
  scanWorkflowVersionForRepo,
} from './scanners/workflowVersionScanner.js'
import { scanOverrideSentinelsForRepo } from './scanners/overrideSentinelScanner.js'
import type {
  CoverageResponse,
  CoverageRow,
  CoverageColumnState,
} from '@agenticapps/dashboard-shared'
import { homedir } from 'node:os'
import { join } from 'node:path'

export interface ScanCoverageOptions {
  sourcecodeRootOverride?: string
  gitnexusHomeOverride?: string
  migrationsDirOverride?: string
}

export async function scanCoverage(opts: ScanCoverageOptions = {}): Promise<CoverageResponse> {
  // Step 1: Discover repos (synchronous — readdir-based)
  const repos = discoverRepos(opts.sourcecodeRootOverride)

  // Step 2 & 3: One-shot singleton reads (gitnexus global + workflow head)
  const gnGlobal = scanGitNexusGlobal(opts.gitnexusHomeOverride)
  const workflowHead = readWorkflowHeadVersion(opts.migrationsDirOverride)

  // Step 4: Fan out 5 per-repo scanners in parallel for each repo.
  // 5 scanners × ~45 repos ≈ 225 stat calls. Sub-second with Promise.all parallelism.
  const sourcecodeRoot = opts.sourcecodeRootOverride ?? join(homedir(), 'Sourcecode')
  const rows: CoverageRow[] = await Promise.all(
    repos.map(async (r) => buildRow(r, gnGlobal, workflowHead, sourcecodeRoot))
  )

  // Step 5 & 6: Assemble + sort
  rows.sort((a, b) => (a.family === b.family ? a.repo.localeCompare(b.repo) : a.family.localeCompare(b.family)))

  return {
    schemaVersion: 1,
    generatedAtIso: new Date().toISOString(),
    gitNexusInstalled: gnGlobal.installed,
    workflowHeadVersion: workflowHead,
    rows,
  }
}

async function buildRow(
  repo: DiscoveredRepo,
  gnGlobal: ReturnType<typeof scanGitNexusGlobal>,
  workflowHead: string | null,
  sourcecodeRoot: string,
): Promise<CoverageRow> {
  const familyRoot = join(sourcecodeRoot, repo.family)
  const claudeMd = scanClaudeMd(repo.absPath)
  const gitNexus = rateGitNexusRepo(gnGlobal, repo.absPath)
  const wiki = scanWikiForFamily(familyRoot, repo.name)
  const workflowVersion = scanWorkflowVersionForRepo(repo.absPath, workflowHead)
  const overrides = scanOverrideSentinelsForRepo(repo.absPath)

  return {
    family: repo.family,
    repo: repo.name,
    absPath: repo.absPath,
    claudeMd: toColumn(claudeMd.state, claudeMd.via === 'AGENTS.md' ? 'via AGENTS.md' : undefined),
    gitNexus: toColumn(gitNexus.state, gitNexus.daysSinceIndex !== undefined ? `${gitNexus.daysSinceIndex}d ago` : undefined, gitNexus.daysSinceIndex),
    wiki: toColumn(wiki.state, wiki.hint ?? (wiki.daysSinceCompile !== undefined ? `${wiki.daysSinceCompile}d ago` : undefined), wiki.daysSinceCompile),
    workflowVersion: toColumn(workflowVersion.state, workflowVersion.installedVersion ?? workflowVersion.detail),
    overrideCount: overrides.length,
    overrides: overrides.map(o => ({ phaseSlug: o.phaseSlug, sinceIso: o.sinceIso, source: o.source })),
  }
}

function toColumn(
  state: CoverageColumnState['state'],
  label?: string,
  daysSince?: number,
): CoverageColumnState {
  const result: CoverageColumnState = { state }
  if (label !== undefined) result.label = label
  if (daysSince !== undefined) result.daysSince = daysSince
  return result
}
```
