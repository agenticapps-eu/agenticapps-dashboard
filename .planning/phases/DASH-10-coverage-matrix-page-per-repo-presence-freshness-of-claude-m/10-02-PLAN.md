import { readdirSync, existsSync } from 'node:fs'
import { homedir } from 'node:os'
import { join } from 'node:path'
import { parseFrontmatter } from '../skillsScan.js'

const MIGRATIONS_DIR_DEFAULT = join(homedir(), 'Sourcecode', 'agenticapps', 'claude-workflow', 'migrations')

export function readWorkflowHeadVersion(migrationsDirOverride?: string): string | null {
  const dir = migrationsDirOverride ?? MIGRATIONS_DIR_DEFAULT
  if (!existsSync(dir)) return null
  let entries: string[]
  try {
    entries = readdirSync(dir)
      .filter((f) => /^\d{4}-.+\.md$/.test(f))
      .sort()
      .reverse()
  } catch { return null }
  for (const name of entries) {
    const fm = parseFrontmatter(join(dir, name))
    if (!fm) continue
    const toVersion = (fm as any).to_version as string | undefined
    if (typeof toVersion === 'string' && toVersion.trim()) return toVersion.trim()
  }
  return null
}

  state: 'fresh' | 'stale' | 'missing' | 'not-applicable'

  headVersion?: string
  detail?: 'version-unknown' | 'behind' | 'ahead' | 'equal' | 'skill-missing'
}

  repoAbsPath: string

  resolve: PathResolver
}

export function scanWorkflowVersionForRepo(input: ScanWorkflowInput): WorkflowRepoState {
  const { repoAbsPath, head, resolve } = input
  for (const p of CANDIDATE_PATHS(repoAbsPath)) {
    // CODEX HIGH-3: canonicalise via resolver before reading
    let canonical: string
    try {
      canonical = resolve(p, { allowedNames: ['SKILL.md'], roots: [skillRootOf(repoAbsPath)] })
    } catch { continue }
    if (!existsSync(canonical)) continue
    const fm = parseFrontmatter(canonical)
    if (!fm) continue
    // CODEX MED-12: identity check — name MUST equal 'agentic-apps-workflow'
    const name = (fm as any).name as string | undefined
    if (typeof name !== 'string' || name.trim() !== 'agentic-apps-workflow') continue
    const ver = (fm as any).version as string | undefined
    if (typeof ver !== 'string' || !ver.trim()) {
      // Pitfall 3 — dashboard's own case
      return { kind: 'workflow', state: 'stale', installedVersion: null, headVersion: head, detail: 'version-unknown' }
    }
    if (!head) return { kind: 'workflow', state: 'fresh', installedVersion: ver.trim(), headVersion: null, detail: 'ahead' }
    const cmp = compareSemver(ver.trim(), head)
    if (cmp === 0) return { kind: 'workflow', state: 'fresh', installedVersion: ver.trim(), headVersion: head, detail: 'equal' }
    if (cmp < 0) return { kind: 'workflow', state: 'stale', installedVersion: ver.trim(), headVersion: head, detail: 'behind' }
    return { kind: 'workflow', state: 'fresh', installedVersion: ver.trim(), headVersion: head, detail: 'ahead' }
  }
  return { kind: 'workflow', state: 'missing', installedVersion: null, headVersion: head, detail: 'skill-missing' }
}

function compareSemver(a: string, b: string): -1 | 0 | 1 {
  const pa = a.split('.').map(Number), pb = b.split('.').map(Number)
  for (let i = 0; i < 3; i++) {
    const x = pa[i] ?? 0, y = pb[i] ?? 0
    if (x < y) return -1
    if (x > y) return 1
  }
