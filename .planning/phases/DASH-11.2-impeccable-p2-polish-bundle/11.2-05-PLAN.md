<objective>
Convert the search `<input>` in CoverageToolbar from an uncontrolled (`defaultValue={search}`) component to a controlled component using the hybrid mirror-state + prop-sync pattern (D-11.2-13). This makes URL back-button navigation re-seed the input correctly (today the input is stuck on whatever the user last typed even after the URL changes).

Purpose: Close P3 #5 from `11.1-IMPECCABLE.md`. The current `defaultValue={search}` only seeds the input on the very first render — subsequent URL changes (via browser back, deep links, programmatic navigation) leave the input out of sync. The mirror-state pattern fixes this without losing Phase 10's deliberate decision to encapsulate the 200ms debounce inside the toolbar.

Output: 1 modified component + 1 modified test file. Roughly 6 lines of code change in CoverageToolbar.tsx (add useState + useEffect, replace `defaultValue` with `value`, capture value via state setter on change).

Implements: D-11.2-13 (hybrid controlled with prop sync; debounce stays in toolbar).

Wave 1 parallel-safe: `CoverageToolbar.tsx` is exclusively owned by this plan — no other Phase 11.2 plan touches it.
</objective>

import React, { useEffect, useRef, useState } from 'react'

export function CoverageToolbar({ filter, search, onFilterChange, onSearchChange }) {
  const debounceRef = useRef<ReturnType<typeof setTimeout> | null>(null)
  const [inputValue, setInputValue] = useState(search)

  useEffect(() => {
    setInputValue(search)
  }, [search])

  function handleSearchChange(e: React.ChangeEvent<HTMLInputElement>) {
    const value = e.target.value
    setInputValue(value)
    if (debounceRef.current) clearTimeout(debounceRef.current)
    debounceRef.current = setTimeout(() => {
      onSearchChange(value)
    }, 200)
  }
