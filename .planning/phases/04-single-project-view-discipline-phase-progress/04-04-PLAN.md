       /**
        * Phase 4 SPA query hooks for the per-project detail panels.
        *
        * D-4-02: 5s polling cadence, 5s daemon memo, refetchIntervalInBackground:false
        * (verbatim from Phase 3 D-02/D-03 — established Phase 3 D-02 pattern).
        *
        * Schema drift: each hook calls apiFetch -> parseOrDrift. On drift it throws
        * `Error('schema_drift:<path>')`. The QueryCache.onError handler in
        * lib/queryClient.ts surfaces this as a per-panel <SchemaDriftState> via
        * the QueryBridge wired in main.tsx (Phase 2 D-09 pattern).
        */
       import { useQuery } from '@tanstack/react-query'
       import {
         CommitmentBlockResponseSchema, type CommitmentBlockResponse,
         ObservationsRecentResponseSchema, type ObservationsRecentResponse,
         DisciplineResponseSchema, type DisciplineResponse,
         PhaseProgressResponseSchema, type PhaseProgressResponse,
         SecurityResponseSchema, type SecurityResponse,
       } from '@agenticapps/dashboard-shared'

       import { apiFetch } from './api.js'

       const POLL_MS = 5_000
       const DEFAULT_OBS_LIMIT = 20

       export function useCommitment(id: string | null) {
         return useQuery({
           queryKey: ['commitment', id] as const,
           queryFn: async (): Promise<CommitmentBlockResponse> => {
             const result = await apiFetch(`/api/projects/${id}/commitment`, CommitmentBlockResponseSchema)
             if (!result.ok) throw new Error(`schema_drift:${result.drift.path}`)
             return result.data
           },
           enabled: id !== null,
           staleTime: POLL_MS,
           refetchInterval: POLL_MS,
           refetchIntervalInBackground: false,
         })
       }

       export function useObservations(id: string | null, limit: number = DEFAULT_OBS_LIMIT) {
         return useQuery({
           queryKey: ['observations', id, limit] as const,
           queryFn: async (): Promise<ObservationsRecentResponse> => {
             const result = await apiFetch(
               `/api/projects/${id}/observations/recent?limit=${limit}`,
               ObservationsRecentResponseSchema,
             )
             if (!result.ok) throw new Error(`schema_drift:${result.drift.path}`)
             return result.data
           },
           enabled: id !== null,
           staleTime: POLL_MS,
           refetchInterval: POLL_MS,
           refetchIntervalInBackground: false,
         })
       }

       export function useDiscipline(id: string | null) {
         return useQuery({
           queryKey: ['discipline', id] as const,

           staleTime: POLL_MS,
           refetchInterval: POLL_MS,
           refetchIntervalInBackground: false,
         })
       }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → daemon (HTTPS/HTTP) | Hooks call apiFetch which automatically sends `Authorization: Bearer ${token}` from localStorage pairing. Token never enters URL or query string. |
| TanStack Query cache (browser memory) | Each query cache entry is keyed by `[name, projectId(, limit)]` — no cross-project bleed even if the user switches projects rapidly. |
| Schema validation | parseOrDrift gates every response. Drift throws `Error('schema_drift:...')` so the QueryCache.onError surface (Phase 2 D-09) renders SchemaDriftState in the panel slot rather than spreading malformed data into UI state. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-04-01 | I (Information Disclosure) | Cross-project leak via cache key reuse | mitigate | Every queryKey includes `projectId`. Tests Q11 explicitly assert that two ids produce two cache slots. observations queryKey ALSO includes `limit` so different limits don't share data. |
| T-04-04-02 | T (Tampering) | URL injection via projectId in fetch path | mitigate | projectId comes from TanStack Router's path params (urldecoded by the router). The daemon validates by exact-match on registry slugs (Phase 1 D-23: slug regex). A maliciously crafted URL with `..` segments doesn't reach the daemon — the router would 404. |
| T-04-04-03 | T (Tampering) | observations limit URL parameter accepts any string | mitigate | Daemon-side clamp (Plan 03 Task 1: MAX_LIMIT=100, fallback to default on non-numeric). | T-04-04-04 | I (Information Disclosure) | document.title leaking projectId outside the page | accept | Title is browser-local; never network-transmitted. projectId is the user's own slug from their own registry — no cross-tenant exposure. |
