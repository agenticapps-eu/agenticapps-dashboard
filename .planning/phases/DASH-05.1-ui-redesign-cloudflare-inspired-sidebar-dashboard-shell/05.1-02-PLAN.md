<router_v2_pattern>
<!-- RESEARCH Pattern 3 — pathless layout via id field. -->
<!-- VERIFIED against TanStack Router v1.169 (in workspace). -->

The current router.tsx mounts AppShell at rootRoute. ALL six routes share that shell — including `/onboarding` and `/pair`, which is a bug per D-5.1-03 (pre-paired surfaces should NOT show paired-session chrome). The V2 route shape both fixes the bug AND introduces the pathless layout.

Two route trees built side-by-side, flag selects which one is registered:

```typescript
const useV2 = import.meta.env.VITE_APPSHELL_V2 === '1'

// Legacy tree (default — KEPT through Waves 1-4)
const legacyRouteTree = rootRoute.addChildren([
  indexRoute, onboardingRoute, pairRoute, settingsRoute, helpRoute, projectsIdRoute,
] as AnyRoute[])

// V2 tree (opt-in via .env.local)
const appShellLayoutRoute = createRoute({
  getParentRoute: () => rootRoute,
  id: '_appshell',
  component: AppShellV2,
})
const indexRouteV2 = createRoute({
  getParentRoute: () => appShellLayoutRoute,  // ← V2 nests under pathless layout
  path: '/',
  beforeLoad: () => { /* same as legacy indexRoute */ },
}).lazy(...)
// ... settingsRouteV2, helpRouteV2, projectsIdRouteV2 likewise nest under appShellLayoutRoute
const onboardingRouteV2 = createRoute({
  getParentRoute: () => rootRoute,  // ← stays at root, NO shell
  path: '/onboarding',
}).lazy(...)
const pairRouteV2 = createRoute({
  getParentRoute: () => rootRoute,  // ← stays at root, NO shell
  path: '/pair',
  validateSearch: ...,
  errorComponent: pairErrorComponent,
}).lazy(...)

const v2RouteTree = rootRoute.addChildren([
  appShellLayoutRoute.addChildren([indexRouteV2, settingsRouteV2, helpRouteV2, projectsIdRouteV2] as AnyRoute[]),
  onboardingRouteV2,
  pairRouteV2,
] as AnyRoute[])

const routeTree = useV2 ? v2RouteTree : legacyRouteTree
```

CRITICAL changes vs current router.tsx:
1. `rootRoute` becomes a bare passthrough: `createRootRoute({ component: () => <Outlet /> })` (no shell). Required because in V2 the shell lives ONE level down at `appShellLayoutRoute`. In legacy mode, rootRoute mounting AppShell stays unchanged — different rootRoute exports for legacy vs V2 are NOT needed because `<Outlet />` is what AppShell already renders inside its <main>; the cosmetic difference is inert when the legacy tree is registered. WAIT: this would break legacy. Solution: declare TWO rootRoutes is not possible (createRootRoute is a singleton per router). So:
   - rootRoute KEEPS `component: AppShell` for legacy mode.
   - V2 mode CANNOT also nest under that AppShell because then V2 routes would have AppShell wrapping AppShellV2 (double shell).
   - Therefore: when useV2 is true, rootRoute's component is set to `() => <Outlet />` (bare). When useV2 is false, rootRoute's component is `AppShell`. This switch happens at module scope, before createRouter runs. Concretely:
     ```typescript
     const rootRoute = createRootRoute({
       component: useV2 ? () => <Outlet /> : AppShell,
     })
     ```

2. `import { Outlet } from '@tanstack/react-router'` added.

3. The flag is read ONCE at module scope (Pitfall 4: strict equality `=== '1'`).

4. `.env.example` documents the flag with a one-line note.

    SidebarItem.tsx (RESEARCH Code Example 2):
    ```typescript
    import React from 'react'
    import { Link, useMatchRoute } from '@tanstack/react-router'

    export interface SidebarItemProps {
      to: string
      params?: Record<string, string>
      icon: React.ReactNode
      label: string
    }

    export function SidebarItem({ to, params, icon, label }: SidebarItemProps): React.JSX.Element {
      const matchRoute = useMatchRoute()
      const isActive = !!matchRoute({ to, params: params ?? {}, fuzzy: false } as Parameters<typeof matchRoute>[0])

      const stateClasses = isActive
        ? 'bg-accent-bg-strong text-white'
        : 'text-text-primary hover:bg-accent-bg'

      return (
        <Link
          to={to as Parameters<typeof Link>[0]['to']}
          {...(params ? { params } as Parameters<typeof Link>[0] : {})}
          className={`flex items-center gap-3 rounded-md px-3 py-2 text-md font-medium ${stateClasses} focus:outline-none focus-visible:ring-2 focus-visible:ring-accent`}
        >
          <span aria-hidden="true" className="shrink-0">{icon}</span>
          <span className="truncate">{label}</span>
        </Link>
      )
    }
    ```

    Breadcrumb.tsx — derive from useMatches():
    ```typescript
    import { Link, useMatches } from '@tanstack/react-router'
    import { ChevronRight } from 'lucide-react'

    interface Crumb { label: string; to: string }

    export function Breadcrumb(): React.JSX.Element {
      const matches = useMatches()
      const last = matches[matches.length - 1]
      const path = last?.fullPath ?? '/'

      const crumbs: Crumb[] = [{ label: 'All Projects', to: '/' }]
      if (path.startsWith('/projects/')) {
        const projectId = (last?.params as { projectId?: string } | undefined)?.projectId
        if (projectId) crumbs.push({ label: projectId, to: '/' })
      } else if (path === '/settings') {
        crumbs[0] = { label: 'Settings', to: '/settings' }
      } else if (path === '/help') {
        crumbs[0] = { label: 'Help', to: '/help' }
      }

      return (
        <nav aria-label="Breadcrumb" className="flex items-center gap-2 text-sm text-text-secondary">
          {crumbs.map((c, i) => (
            <span key={i} className="flex items-center gap-2">
              {i > 0 && <ChevronRight size={14} aria-hidden="true" className="text-text-tertiary" />}
              {i === crumbs.length - 1 ? (
                <span className="font-semibold text-text-primary">{c.label}</span>
              ) : (
                <Link to={c.to as '/'} className="hover:text-accent focus:outline-none focus-visible:ring-2 focus-visible:ring-accent">{c.label}</Link>
              )}
            </span>
          ))}
        </nav>
      )
    }
    ```
